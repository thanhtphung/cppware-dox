129 //! Recursively apply callback to directory entries. Children first.
130 //! The callback should return true to continue iterating and should
131 //! return false to abort iterating. Return false if the callback
132 //! aborted the iterating. Return true otherwise.
133 //!
134 bool Directory::applyChildFirst(cb0_t cb, void* arg) const
135 {
136  String files;
137  String subdirs;
138  list(files, subdirs);
139  if (files.empty() && subdirs.empty())
140  {
141  return true;
142  }
143 
144  // Children first. Return as soon as iterating is aborted.
145  String subdir;
146  Directory child;
147  DelimitedTxt txt(subdirs, false, DELIM);
148  for (String path(path_);
149  txt.next(subdir) && (!txt.trimLine(subdir).empty());
150  path.truncate(path_.length()))
151  {
152  child.setPath(path += subdir);
153  if (!child.applyChildFirst(cb, arg))
154  {
155  return false;
156  }
157  }
158 
159  // Then subdirs. Return as soon as iterating is aborted.
160  txt.reset();
161  while (txt.next(subdir) && (!txt.trimLine(subdir).empty()))
162  {
163  if (!cb(arg, *this, subdir))
164  {
165  return false;
166  }
167  }
168 
169  // Then files. Return as soon as iterating is aborted.
170  String file;
171  txt.setTxt(files, false);
172  while (txt.next(file) && (!txt.trimLine(file).empty()))
173  {
174  if (!cb(arg, *this, file))
175  {
176  return false;
177  }
178  }
179 
180  // Return true to indicate iterating was not aborted.
181  return true;
182 }

257 {

260  if (childName.byteSize() > findArg.dotExtLength)
261  {
262  const char DOT = '.';

271  }

273  return keepGoing;
