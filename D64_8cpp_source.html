36 // Return 1 if wrapping seems valid.
37 // Return 0 if there seems to be no wrapping.
38 // Return -1 if wrapping seems invalid.
39 int parseWrapping(const appkit::String& s)
40 {
41  int rc;
42  const char* p = s.ascii();
43  switch (p[0])
44  {
45  case '(': rc = (s[s.byteSize() - 2] == ')')? 1: -1; break;
46  case '[': rc = (p[s.byteSize() - 2] == ']')? 1: -1; break;
47  case '{': rc = (p[s.byteSize() - 2] == '}')? 1: -1; break;
48  case '<': rc = (p[s.byteSize() - 2] == '>')? 1: -1; break;
49  case '\'':
50  case '"':
51  case '|': rc = (p[s.byteSize() - 2] == p[0])? 1: -1; break;
52  default: rc = 0; break;
53  }
54 
55  return rc;
56 }

136  double v;

143
