104 //! Retrieve the next column. Return true if there's one. Otherwise, return
105 //! false and an empty column (if row is empty or if there's no more columns).
106 //! The first invocation after construction or reset() will return the first
107 //! column. If doDequoteCol is true, the returned column is also trimmed and
108 //! dequoted. For a single quoted column, escape sequence resolution can be

112  bool ok = doDequoteCol? dequoteNext(col, unescapeSingleQuotedString): getNext(col);

123 inline const String& CsvRow::trimCol(String& col) const
124 {
125  size_t n = col.byteSize() - 1; //index of last byte in front of terminating null
126  if ((n > 0) && (col.raw()[n - 1] == static_cast<syskit::utf8_t>(delim_))) col.truncate(col.length() - 1);
