464 //
465 // news-watch [--maxheadlines=xxx] [--allbutthese] [subject...]
466 // [--maxsecs=xxx ]
467 //
468 bool NewsCmd::doWatch(const CmdLine& req)
469 {
470 
471  // Identify news subjects of interest.
472  const NewsAnchor& newsAnchor = *NewsAnchor::instance();
473  StringVec* subjectNames = new StringVec;
474  if (req.numArgs() > 1)
475  {
476  const StringVec& argVec = req.argVec();
477  size_t startAt = 1;
478  size_t itemCount = argVec.numItems() - 1;
479  if (req.opt("allbutthese")) //watch all but these news subjects
480  {
481  newsAnchor.apply(getSubjects, subjectNames);
482  *subjectNames = *subjectNames - StringVec(argVec, startAt, itemCount);
483  }
484  else //watch only these news subjects
485  {
486  *subjectNames = StringVec(argVec, startAt, itemCount);
487  }
488  }
489  else //watch all
490  {
491  newsAnchor.apply(getSubjects, subjectNames);
492  }
493 
494  unsigned int maxHeadlines = U32(req.opt("maxheadlines"), NEWS_WATCH_HEADLINES);
495  unsigned int maxSecs = U32(req.opt("maxsecs"), NEWS_WATCH_SECS);
496 
497  Semaphore sem(0U);
498  NewsWatch* newsWatch = new NewsWatch(&sem, &req, maxHeadlines);
499  NewsSubject::attachByNames(newsWatch, subjectNames);
500  unsigned int waitTimeInMsecs = maxSecs * MSECS_PER_SEC;
501  sem.decrement(waitTimeInMsecs);
502  NewsSubject::detachByNames(newsWatch, subjectNames);
503  delete subjectNames;
504  newsWatch->rmRef();
505 
506  String rsp("Done.");
507  respond(req, rsp);
508  bool cmdIsValid = true;
509  return cmdIsValid;
510 }
