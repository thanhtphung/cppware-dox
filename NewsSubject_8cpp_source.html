172 bool NewsSubject::doAttach1(void* arg, NewsSubject* source)
173 {
174  doAttach1Arg_t* p = static_cast<doAttach1Arg_t*>(arg);
175  bool keepGoing = (strcmp(p->name, source->name()) != 0);
176  if (!keepGoing)

178  source->attach(p->observer); //failure indicates already attached

186 bool NewsSubject::doAttachN(void* arg, NewsSubject* source)
187 {
188  bool keepGoing = true;
189  doAttachNArg_t* p = static_cast<doAttachNArg_t*>(arg);
190  String name(source->name());
191  if (p->sortedNames->search(name))
192  {
193  source->attach(p->observer); //failure indicates already attached
194  p->attached->add(name);
195  if (p->attached->numItems() == p->sortedNames->numItems())
196  {
197  keepGoing = false;
198  }
199  }
200 
201  return keepGoing;
202 }

267 //!
268 //! Mark subject as updated and return current time as latest update time. Also associate
269 //! this update with given summary. Observers will be notified ASAP via notify() and
270 //! Observer::onUpdate(). Notifications are performed in a separate news anchor thread.
271 //! To clarify, everytime a news subject is updated, a story is created via createStory().
272 //! This story is then broadcasted to its audience by a dedicated news anchor. As a result,
273 //! multiple stories within one news subject can be in the queue waiting to be broadcasted.
274 //!
275 unsigned long long NewsSubject::update(const String& summary)
276 {
277  unsigned long long updateTime = Observable::update(summary);
278  putStory(updateTime, summary);
279  newsAnchor_->observe(this);
280  return updateTime;
281 }
