17 class Utf16
18  //!
19  //! A class representing a UTF16 character. Natively, a UTF16 character has a
20  //! 32-bit value with range 0x000000..0x00d7ff,0x00e000..0x10ffff. When encoded,
21  //! a 16-bit value uses one unsigned short, and a larger value uses two unsigned
22  //! shorts. Construct an instance to encode/decode a UTF16 character. Refer to
23  //! RFC 2781 for more information. The www.unicode.org FAQs do mention these
24  //! invalid 34 values: 0x00fdd0..0x00fdef,0x00fffe..0x00ffff. However, there
25  //! seems to be discrepancies regarding these values in various RFCs and various
26  //! implementations. Those 34 values are considered valid in this class. Where
27  //! byte-ordering is a factor, data is assumed to have native-endian byte-ordering.
28  //! Example:
29  //!\code
30  //! Utf16 c0(0x54321U);
31  //! utf16_t seq[Utf16::MaxSeqLength];
32  //! unsigned int seqLength = c0.encode(seq);
33  //! Utf16 c1;
34  //! if ((c1.decode(seq) != seqLength) || (c1.asU32() != c0.asU32()))
35  //! {
36  //! assert("coding error" == 0);
37  //! }
38  //!\endcode
39  //!
40 {

161 //! Decode given UTF16 byte sequence (either 1 or 2 unsigned shorts starting
162 //! at seq). If successful, update instance with decoded value and return
163 //! the number of unsigned shorts decoded. Otherwise (i.e., given UTF16 byte
164 //! sequence is malformed), return zero.
165 inline unsigned int Utf16::decode(const utf16_t seq[MaxSeqLength])
166 {
167  if ((seq[0] < HiHalfMin) || (seq[0] > LoHalfMax))
168  {
169  c_ = seq[0];
170  return 1;
171  }
172  if ((seq[0] <= HiHalfMax) && (seq[1] >= LoHalfMin) && (seq[1] <= LoHalfMax))
173  {
174  c_ = Anchor + (((seq[0] & 0x3ffU) << 10) | (seq[1] & 0x3ffU));
175  return 2;
176  }
177  return 0;
178 }

180 //! Decode given valid UTF16 byte sequence (either 1 or 2 unsigned shorts
181 //! starting at validSeq). Save the effective value in c. Return the
182 //! number of unsigned shorts decoded.
183 inline unsigned int Utf16::decodeValidSeq(const utf16_t* validSeq, utf32_t& c)
184 {
185  return ((validSeq[0] < HiHalfMin) || (validSeq[0] > LoHalfMax))?
186  ((c = validSeq[0]), 1):
187  ((c = Anchor + (((validSeq[0] & 0x3ffU) << 10) | (validSeq[1] & 0x3ffU))), 2);
188 }

190 //! Encode value into a UTF16 byte sequence. A 16-bit value uses one unsigned
191 //! short, and a larger value uses two unsigned shorts. Save results in seq.
192 //! Assume seq is a buffer large enough for this value. Return the number of
193 //! unsigned shorts utilized in the encoding.
194 inline unsigned int Utf16::encode(utf16_t seq[MaxSeqLength]) const
195 {
196  if (c_ <= 0xffffU)
197  {
198  seq[0] = static_cast<unsigned short>(c_);
199  return 1;
200  }
201  unsigned int delta = c_ - Anchor;
202  seq[0] = static_cast<unsigned short>(HiHalfMin | (delta >> 10));
203  seq[1] = static_cast<unsigned short>(LoHalfMin | (delta & 0x3ffU));
204  return 2;
205 }

239 //! Convert given valid UTF16 byte sequence (either 1 or 2 unsigned shorts
240 //! starting at validSeq) into its effective value. Return the effective
241 //! value. Behavior is unpredictable if given sequence is invalid.
242 inline utf32_t Utf16::convertValidSeq(const utf16_t* validSeq)
243 {
244  return ((validSeq[0] < HiHalfMin) || (validSeq[0] > LoHalfMax))?
245  (validSeq[0]):
246  (Anchor + (((validSeq[0] & 0x3ffU) << 10) | (validSeq[1] & 0x3ffU)));
