161 bool CmdCmd::doShowLog(const CmdLine& req)
162 {
163 
164  // Log file is not seekable.
165  // Command likely being run via sprint().
166  std::FILE* log = Cmd::log(req);
167  int posInCurLog = std::ftell(log);
168  if (posInCurLog < 0)
169  {
170  String rsp("None.");
171  respond(req, rsp);
172  }
173 
174  // Show command log head if the head option is specified. Show command log tail otherwise.
175  // Disable response logging for this command to minimize confusion in the log. If zero is
176  // given as a head or tail option, interpret it as a one.
177  else
178  {
179  bool logReq = sysIo_->reqLoggingIsEnabled();
180  bool logRsp = sysIo_->rspLoggingIsEnabled();
181  bool flushLogEntries = sysIo_->logEntriesAreFlushed();
182  sysIo_->setLogMode(logReq, false /*logRsp*/, flushLogEntries);
183 
184  String optK("head");
185  const String* optV = req.opt(optK);
186  if (optV == 0)
187  {
188  optK = "tail";
189  unsigned int defaultV = 10;
190  unsigned int cmdCount = U32(req.opt(optK), defaultV);
191  tailLog(req, log, posInCurLog, cmdCount? cmdCount: 1);
192  }
193  else
194  {
195  unsigned int cmdCount = U32(*optV);
196  headLog(req, log, posInCurLog, cmdCount? cmdCount: 1);
197  }
198 
199  sysIo_->setLogMode(logReq, logRsp, flushLogEntries);
200  }
201 
202  bool cmdIsValid = true;
203  return cmdIsValid;
204 }

291 bool CmdCmd::touchupCmdLog(void* arg, const char* line, size_t length)
292 {
293 
294  // These lines are constant at the callback interface, but they reside in the
295  // buffer referenced to by DelimitedTxt, and they can be touched up as needed.
296  // To minimize confusion in the cmd-showlog output, replace vertical lines with
297  // equals signs. Abort the iterating after about cmdCount commands have been
298  // seen. Assume a command echo in the log files starts with a vertical line and
299  // is at least 30 characters long.
300  bool keepGoing = true;
301  if ((length > 0) && (line[0] == VERTICAL_LINE))
302  {
303 
304  char* p = const_cast<char*>(line);
305  p[0] = EQUALS_SIGN;
306  for (size_t i = 1; i < length; ++i)
307  {
308  if (p[i] == VERTICAL_LINE)
309  {
310  p[i] = EQUALS_SIGN;
311  }
312  }
313 
314  if (length > 30)
315  {
316  cmdShowLogArg_t* p = static_cast<cmdShowLogArg_t*>(arg);
317  if (--p->cmdCount <= 0)
318  {
319  p->rsp = line;
320  keepGoing = false;
321  }
322  }
323  }
324 
325  return keepGoing;
326 }

348 // Show the head end of the command log including the current open log file.
349 // Show at most cmdCount commands.
350 //
351 void CmdCmd::headLog(const CmdLine& req, std::FILE* log, int posInCurLog, int cmdCount)
352 {
353  const char* name = (reqPort(req) == sysIo_->addr(0).port())? "-sysio-a": "-sysio-b";
354  char lo = sysIo_->logPathLo();
355  char hi = sysIo_->logPathHi();
356  String newestPath(LogPath::newestPath(name, lo, hi));
357  String oldestPath(LogPath::oldestPath(name, lo, hi));
358  headLog(req, newestPath, oldestPath, lo, hi, cmdCount);
359 
360  // Need to show log entries from current log file now as the
361  // older log files do not have all of the requested enries.
362  if (cmdCount > 0)
363  {
364 
365  // Read the whole current log into a null-terminated buffer. It's likely only
366  // a small head-end portion is needed, but the simplified logic is more important
367  // for this not-frequently-used nice-to-have troubleshooting capability.
368  std::fseek(log, 0, SEEK_SET);
369  int bufSize = posInCurLog;
370  char* buf = new char[bufSize];
371  std::fread(buf, bufSize, 1, log);
372  std::fseek(log, posInCurLog, SEEK_SET);
373 
374  // To minimize confusion in the cmd-showlog output, replace vertical lines with equals signs.
375  bool makeCopy = false;
376  DelimitedTxt txt(buf, bufSize, makeCopy);
377  cmdShowLogArg_t arg(buf + bufSize, cmdCount + 1);
378  txt.applyLoToHi(touchupCmdLog, &arg);
379 
380  size_t rspLength = (arg.rsp - buf);
381  respond(req, buf, rspLength);
382  delete[] buf;
383  }
384 
385  const char* rsp = "";
386  respond(req, rsp);
387 }

394 void CmdCmd::headLog(const CmdLine& req, const String& newestPath, const String& curPath, char lo, char hi, int& cmdCount)
395 {
396  bool readOnly = true;
397  bool skipBom = true;
398  MappedTxtFile logFile(curPath.widen(), readOnly, skipBom);
399  if (!logFile.isOk())
400  {
401  return;
402  }
403 
404  // To minimize confusion in the cmd-showlog output, replace vertical lines with equals signs.
405  size_t bufSize = static_cast<size_t>(logFile.imageSize());
406  char* buf = new char[bufSize];
407  memcpy(buf, logFile.image(), bufSize);
408  bool makeCopy = false;
409  DelimitedTxt txt(buf, bufSize, makeCopy);
410  cmdShowLogArg_t arg(buf + bufSize, cmdCount + 1);
411  txt.applyLoToHi(touchupCmdLog, &arg);
412 
413  int cmdsShown = cmdCount - arg.cmdCount + 1;
414  cmdCount -= cmdsShown;
415  size_t rspLength = (arg.rsp - buf);
416  respond(req, buf, rspLength);
417  delete[] buf;
418 
419  // Recursively if necessary.
420  if (cmdCount > 0)
421  {
422  String newerPath(LogPath::newerPath(curPath, lo, hi));
423  if (newerPath != newestPath)
