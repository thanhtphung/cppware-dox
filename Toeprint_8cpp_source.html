108 unsigned int Toeprint::checksum(const char* key)
109 {
110  unsigned int sum = 0;
111  unsigned int packed[3] = {0, 0, 0};
112  unsigned int i = 0;
113  for (const char* s = key; *s; ++i, ++s)
114  {
115  unsigned int c = *s;
116  unsigned int k64 = s_key[c];
117  switch (i)
118  {
119  case 0:
120  packed[0] = k64;
121  //--packed[1] already 0;
122  //--packed[2] already 0;
123  break;
124  case 1:
125  case 2:
126  case 3:
127  case 4:
128  packed[0] = (packed[0] << 6) | k64;
129  break;
130  case 5:
131  packed[0] = (packed[0] << 2) | (k64 >> 4);
132  packed[1] = k64 & 0x0fU;
133  break;
134  case 6:
135  packed[1] = (packed[1] << 4) | k64;
136  break;
137  case 7:
138  case 8:
139  case 9:
140  packed[1] = (packed[1] << 6) | k64;
141  break;
142  case 10:
143  packed[1] = (packed[1] << 4) | (k64 >> 2);
144  packed[2] = k64 & 0x03U;
145  break;
146  case 11:
147  case 12:
148  case 13:
149  case 14:
150  case 15:
151  packed[2] = (packed[2] << 6) | k64;
152  break;
153  default: //16
154  sum ^= packed[0] ^ packed[1] ^ packed[2];
155  packed[0] = k64;
156  packed[1] = 0;
157  packed[2] = 0;
158  i = 0;
159  break;
160  }
161  }
162 
163  if (i != 1)
164  {
165  sum ^= packed[0] ^ packed[1] ^ packed[2];
166  }
167 
168  sum = bswap32(sum);
169  return sum;
170 }
