24 //!
25 //! Construct a buffer pool given its configuration. The configuration
26 //! is a string of triplets delimited by a semicolon. Each triplet is a
27 //! string delimited by a colon. Each triplet contains characteristics
28 //! of an individual buffer arena, namely 1) buffer size, 2) initial
29 //! capacity, and 3) the growth factor. Triplets with invalid or duplicate
30 //! buffer sizes are ignored. For example, the "4:256:256;128:0:32;"
31 //! configuration string specifies two buffer arenas. The first one has
32 //! buffer size 4 and initial capacity of 256 buffers, and it grows
33 //! linearly by 256 buffers as needed. The second one has buffer size
34 //! 128 and initial capacity of zero, and it grows by 32 buffers as
35 //! needed. A special configuration of "0:0:0;" can be used to disable
36 //! the buffer pool. For a disabled pool, buffers come from the default
37 //! c++ heap.
38 //!
39 BufPool::BufPool(const char* config)
40 {
41  const char* cf;
42  if (config == 0)
43  {
44  cf = "";
45  maxBufSize_ = MaxBufSize;
46  }
47  else
48  {
49  cf = config;
50  maxBufSize_ = (strcmp(config, "0:0:0;") == 0)? 0: MaxBufSize;
51  }
52 
53  construct(cf);
54 }

185 //
186 // Construct pool using given configuration.
187 //
188 void BufPool::construct(const char* config)
189 {
190  arena_[0] = 0;
191  ss_[0] = 0;
192  for (unsigned int bufSize = 4; bufSize <= maxBufSize_; bufSize += 4)
193  {
194  arena_[bufSize] = 0;
195  ss_[bufSize] = new SpinSection;
196  }
197 
198  // Construct individual arenas as configured.
199  // Ignore invalid/duplicate buffer sizes.
200  char* p1;
201  char* txt0 = syskit::strdup(config);
202  for (char* p0 = txt0; (p1 = strchr(p0, ';')) != 0; p0 = p1 + 1)
203  {
204  *p1 = 0;
205  unsigned int bufSize;
206  unsigned int capacity;
207  int growBy;
208  bool ok = getArenaConfig(p0, bufSize, capacity, growBy);
209  if (ok && (bufSize > 0) && (bufSize <= maxBufSize_) && ((bufSize & 3) == 0) && (arena_[bufSize] == 0))
210  {
211  arena_[bufSize] = new BufArena(bufSize, capacity, growBy);
212  }
213  }
214  delete[] txt0;
215 
216  // Construct default arenas.
217  for (unsigned int bufSize = 4; bufSize <= maxBufSize_; bufSize += 4)
218  {
219  if (arena_[bufSize] == 0)
220  {
221  int growBy = ARENA_SIZE / bufSize;
222  unsigned int capacity = 0;
223  arena_[bufSize] = new BufArena(bufSize, capacity, growBy);
224  }
225  BufArena* arena = arena_[bufSize];
226  arena_[bufSize - 3] = arena;
227  arena_[bufSize - 2] = arena;
228  arena_[bufSize - 1] = arena;
229  SpinSection* ss = ss_[bufSize];
230  ss_[bufSize - 3] = ss;
231  ss_[bufSize - 2] = ss;
232  ss_[bufSize - 1] = ss;
233  }
234 }
