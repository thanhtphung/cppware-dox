174 //! Form command echo used in logging. This echo contains current timestamp and
175 //! the debuggee's address. For example, "|10/25/11 15:46:11|10.0.4.82:1911| ".
176 //!
177 String CmdAgent::formReqEcho(const String& req) const
178 {
179 
180  // Form timestamp (example: "|10/25/11 15:46:11|10.0.4.82:1911| ").
181  struct tm now;
182  time_t now0 = time(0);
183  localtime_s(&now, &now0);
184  char stamp[32 + Paddr::StrLength + 2];
185  size_t size = strftime(stamp, 32, "\n|%c|", &now);
186  size += dbuggee_.toDigits(stamp + size, COLON);
187  stamp[size++] = '|';
188  stamp[size++] = ' ';
189 
190  String echo(stamp, size);
191  echo += req;
192  echo += '\n';
193  return echo;

197 //!
198 //! Ping debuggees at given subnet using given port. Return responsive
199 //! dbuggees. The returned value is a dictionary with responsive debuggee
200 //! addresses (e.g., "10.0.4.82:1911") as keys and corresponding responses
201 //! (e.g., "pong") as values.
202 //!
203 StringDic CmdAgent::broadcast(const String& ping, const Subnet& subnet, unsigned short port)
204 {
205  StringDic pong;
206  UdpClient io(IBUF_SIZE, OBUF_SIZE);
207  Paddr dst(subnet.broadcastAddr(), port);
208  bool ok = io.bind(subnet.addr(), 0) && io.snd(ping.raw(), ping.byteSize(), dst);
209  if (ok)
210  {
211  Paddr src;
212  String k;
213  String v;
214  utf8_t rawRsp[Cmd::MaxRspLength + 1];
215  for (unsigned int timeoutInMsecs = 765;; timeoutInMsecs = 234)
216  {
217  int rspSize = io.rcv(rawRsp, sizeof(rawRsp), src, timeoutInMsecs);
218  if (rspSize > 1)
219  {
220  if (rawRsp[rspSize - 1] == 0)
221  {
222  k = src;
223  v.reset8(rawRsp, rspSize);
224  pong.associate(k, v);
225  }
226  continue;
227  }
228  break;
229  }
230  }
231 
232  return pong;
233 }

236 //!
237 //! Ping debuggees at connected subnets using given port. Return responsive debuggees.
238 //! The returned value is a dictionary with responsive debuggee addresses (e.g.,
239 //! "10.0.4.82:1911") as keys and corresponding responses (e.g., "pong") as
240 //! values. Ignore zero (0.0.0.0) interfaces.
241 //!
242 StringDic CmdAgent::broadcast(const String& ping, unsigned short port)
243 {
244  IpAddrSet pinged;
245  StringDic pong;
246  Subnet subnet;
247  for (unsigned char i = 0, numDevices = IpDevice::numDevices(); i < numDevices; ++i)
248  {
249  const IpDevice* d = IpDevice::instance(i);
250  unsigned int numAddrs = d->numAddrs();
251  for (unsigned int j = 0; j < numAddrs; ++j)
252  {
253  unsigned int addr = d->addr(j);
254  if (addr != 0)
255  {
256  subnet.reset(addr, d->netMask(j));
257  if (!pinged.contains(subnet.broadcastAddr()))
258  {
259  broadcast(ping, subnet, port).mergeInto(pong);
260  pinged.add(subnet.broadcastAddr());
261  }
262  }
263  }
264  }
265 
266  return pong;
267 }

373 //
374 // Show response. Return true if successful. Return false if timed out.
375 //
376 bool CmdAgent::showRsp(String* rspCopy)
377 {
378  if (rspCopy != 0)
379  {
380  rspCopy->reset();
381  }
382 
383  bool ok = true;
384  utf8_t rsp[Cmd::MaxRspLength + 1];
385  for (unsigned int timeoutInMsecs = timeout0_;; timeoutInMsecs = timeout_)
386  {
387  int rspSize = io_->rcv(rsp, sizeof(rsp), timeoutInMsecs);
388  if (rspSize > 0)
389  {
390 
391  // Show partial response, then wait for more.
392  unsigned char eor = rsp[rspSize - 1];
393  if (eor != 0)
394  {
395  bool more = true;
396  onRsp(reinterpret_cast<const char*>(rsp), rspSize, more);
397  saveRsp(rspCopy, rsp, rspSize);
398  continue;
399  }
400 
401  // Show complete response, then return ASAP.
402  else if (rspSize > 1)
403  {
404  bool more = false;
405  onRsp(reinterpret_cast<const char*>(rsp), rspSize - 1, more);
406  saveRsp(rspCopy, rsp, rspSize - 1);
407  }
408  }
409  else
410  {
411  onTimeout();
412  ok = false;
413  }
414  break;
415  }
416 
417  return ok;
418 }
