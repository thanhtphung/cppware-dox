19 //!
20 //! Construct path to a primitive circular log. The circular log resides in the
21 //! application's log directory. The member log files are named using this suffix
22 //! pattern: "<name><index>.txt". The index ranges from lo to hi. As an example,
23 //! LogPath("xxx-",'0','2') is a three-member circular log with these suffixes:
24 //! "xxx-0.txt", "xxx-1.txt", and "xxx-2.txt".
25 //!
26 LogPath::LogPath(const char* name, char lo, char hi):
27 path_()
28 {
29  const App* app = App::instance();
30  String childName(formBasename(app, name));
31  size_t nameIndex = childName.byteSize() - 6; //refers to the x before the file extension "x.txt"
32 
33  // If member log files exist, start with overwriting the oldest one.
34  Directory dir(app->logDir());
35  Directory::Attr attr;
36  Utc oldest;
37  char startAt = lo;
38  for (char c = lo; c <= hi; ++c)
39  {
40  childName.setAscii(nameIndex, c);
41  bool ok = dir.getChildAttr(attr, childName) && (!attr.isDir());
42  if (!ok)
43  {
44  startAt = c;
45  break;
46  }
47  Utc t(attr.writeTime());
48  if (t < oldest)
49  {
50  oldest = t;
51  startAt = c;
52  }
53  }
54 
55  childName.setAscii(nameIndex, startAt);
56  path_ = app->logDir();
57  path_ += childName;
58 
59  lo_ = lo;
60  hi_ = hi;
61  openLog(); //log_
62 }

164 //!
165 //! Return the newer path relative to given seed path (e.g., "/root/bin/Logs/exe-logname-3.txt"
166 //! if seed path is "/root/bin/Logs/exe-logname-2.txt"). Refer to the LogPath constructor for
167 //! info on the parameters (lo, hi) describing the circular log.
168 //!
169 String LogPath::newerPath(const String& seedPath, char lo, char hi)
170 {
171  size_t i = seedPath.byteSize() - 6; //refers to the x before the file extension "x.txt"
172  char cur = seedPath.ascii()[i];
173  char newer = (cur == hi)? (lo): (cur + 1);
174  String path(seedPath);
175  path.setAscii(i, newer);
176  return path;
177 }
