122 bool RoZipped::extractCurItemAs(const wchar_t* path, unsigned long long bytesBeingExtracted) const
123 {
124  unz_file_info64 hdr = {0};
125  unzGetCurrentFileInfo64(zip_, &hdr, 0, 0, 0, 0, 0, 0);
126  unsigned long long size = hdr.uncompressed_size;
127 
128  unzOpenCurrentFile(zip_);
129  bytesBeingExtracted_ = bytesBeingExtracted;
130  bool failIfExists = true;
131  MappedFile file(path, size, failIfExists, mapSize_);
132  unsigned long numMaps = file.numMaps();
133  unsigned long bytesToRead = (numMaps > 1)? file.mapSize(): static_cast<unsigned long>(size);
134  unsigned long long bytesRead = 0;
135  for (unsigned long i = 0; i < numMaps; ++i)
136  {
137  unsigned char* dst = file.map(i);
138  int rc = unzReadCurrentFile(zip_, dst, bytesToRead);
139  bytesRead += rc;
140  if (rc == 0) //extraction being canceled
141  {
142  break;
143  }
144  }
145 
146  bytesBeingExtracted_ = 0;
147  unzCloseCurrentFile(zip_);
148  bool ok = (bytesRead == size);
149  if (ok)
150  {
151  unsigned short dosDate = (hdr.dosDate >> 16);
152  unsigned short dosTime = (hdr.dosDate & 0xffffU);
153  FILETIME localFileTime;
154  DosDateTimeToFileTime(dosDate, dosTime, &localFileTime);
155  FILETIME fileTime;
156  LocalFileTimeToFileTime(&localFileTime, &fileTime);
157  HANDLE h = file.handle();
158  SetFileTime(h, &fileTime, 0, &fileTime);
159  }

162 }
