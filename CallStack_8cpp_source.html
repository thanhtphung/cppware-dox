70 CallStack::CallStack(const _CONTEXT* context, const wchar_t* dumpPath, unsigned long exceptionCode)
71 {
72 
73  // The sym API cannot be used concurrently. An atomic counter is used to prevent
74  // concurrency here. Use of something like a static critical section would make the
75  // callstack capability unavailable before and after static data construction/destruction.
76  static AtomicWord::item_t s_symApiInUse = 0;
77  AtomicWord* symApiInUse = reinterpret_cast<AtomicWord*>(&s_symApiInUse);
78  for (; ++*symApiInUse > 1; --*symApiInUse, Thread::yield());
79 
80  HANDLE process = GetCurrentProcess();
81  HANDLE thread = GetCurrentThread();
82  if (context == 0)
83  {
84  _CONTEXT cr = {0};
85  RtlCaptureContext(&cr);
86  curFrame_ = saveStack(process, thread, &cr);
87  }
88  else
89  {
90  curFrame_ = saveStack(process, thread, context);
91  }
92 
93  --*symApiInUse;
94  exceptionCode_ = exceptionCode;
95  setDumpPath(dumpPath);
96 }

155 const CallStack::Frame* CallStack::saveStack(HANDLE process, HANDLE thread, const _CONTEXT* context)
156 {
157  _CONTEXT cr = *context;
158  STACKFRAME64 frame64 = {0};
159 
160 #if _M_IX86
161  DWORD machineType = IMAGE_FILE_MACHINE_I386;
162  frame64.AddrPC.Offset = cr.Eip;
163  frame64.AddrPC.Mode = AddrModeFlat;
164  frame64.AddrFrame.Offset = cr.Ebp;
165  frame64.AddrFrame.Mode = AddrModeFlat;
166  frame64.AddrStack.Offset = cr.Esp;
167  frame64.AddrStack.Mode = AddrModeFlat;
168 
169 #elif _M_X64
170  DWORD machineType = IMAGE_FILE_MACHINE_AMD64;
171  frame64.AddrPC.Offset = cr.Rip;
172  frame64.AddrPC.Mode = AddrModeFlat;
173  frame64.AddrFrame.Offset = cr.Rsp;
174  frame64.AddrFrame.Mode = AddrModeFlat;
175  frame64.AddrStack.Offset = cr.Rsp;
176  frame64.AddrStack.Mode = AddrModeFlat;
177 
178 #else
179 #error "unsupported architecture"
180 
181 #endif
182 
183  IMAGEHLP_MODULEW64 module = {0};
184  module.SizeOfStruct = sizeof(module);
185 
186  unsigned char symBuf[sizeof(SYMBOL_INFO) + (MAX_SYM_NAME - 1 /*SYMBOL_INFO.Name[1]*/)];
187  SYMBOL_INFO* p = reinterpret_cast<SYMBOL_INFO*>(symBuf);
188  p->SizeOfStruct = sizeof(*p);
189  p->MaxNameLen = MAX_SYM_NAME;
190 
191  IMAGEHLP_LINE64 line = {0};
192  line.SizeOfStruct = sizeof(line);
193 
194  // The UNICODE interfaces (SymInitializeW, SymFromAddrW, and SymGetLineFromAddrW64)
195  // are unavailable on xp-5.1 and server-2003-5.2. The older interfaces (SymInitialize,
196  // SymFromAddr, and SymGetLineFromAddr64) are used to support those environments.
197  // Alternatively, run-time checks via LoadLibraryW() and GetProcAddress() should
198  // be considered when feasible. --ttp
199  Frame* curFrame = new Frame(0);
200  Frame* topFrame = curFrame;
201  BOOL invadeProcess = TRUE;
202  char symPath[MAX_PATH_LENGTH + 1];
203  SymInitialize(process, getSymPath(symPath), invadeProcess);
204  while (StackWalk64(machineType, process, thread, &frame64, &cr, 0, SymFunctionTableAccess64, SymGetModuleBase64, 0))
205  {
206 
207  Frame* frame = new Frame(reinterpret_cast<const void*>(frame64.AddrPC.Offset));
208  topFrame->setNext(frame);
209  topFrame = frame;
210 
211  bool ok = (SymGetModuleInfoW64(process, frame64.AddrPC.Offset, &module) != 0);
212  frame->setModule(ok, module.ModuleName);
213  DWORD64 displacement64 = 0;
214  ok = (SymFromAddr(process, frame64.AddrPC.Offset, &displacement64, p) != 0);
215  frame->setSym(ok, W(p->Name), displacement64);
216  DWORD displacement = 0;
217  ok = (SymGetLineFromAddr64(process, frame64.AddrPC.Offset, &displacement, &line) != 0);
218  frame->setPath(ok, W(line.FileName), line.LineNumber, displacement);
219 
220  if ((frame64.AddrReturn.Offset == 0) || (frame64.AddrReturn.Offset == frame64.AddrPC.Offset))
221  {
222  break;
223  }
224  }
225 
226  SymCleanup(process);
227  const Frame* frame = curFrame->next();
228  curFrame->setNext(0);
229  delete curFrame;
230  return frame;
231 }

335 //! Dump call stack. Show module addresses if showAddrs is true. Show OS and module versions
336 //! if showVersions is true. The call stack dump is appended to the dump file given earlier
337 //! when the call stack was constructed.
338 //!
339 void CallStack::dump(bool showAddrs, bool showVersions) const
340 {
341  wchar_t timestamp[255 + 1];
342  size_t bufSizeInChars = sizeof(timestamp) / sizeof(timestamp[0]);
343  struct tm now;
344  time_t t = time(0);
345  localtime_s(&now, &t);
346  wcsftime(timestamp, bufSizeInChars, L"|%c|", &now);
347 
348  // Show some environment. Example:
349  /*
350  ## |08/31/11 10:32:52| pid=5036 tid=8728
351  cmdLine: "C:\p4c\ipsprobe\vc10\nettools\..\..\bin\vc10-win32\syskit-ut-d.exe"
352  curDir: C:/p4c/ipsprobe/syskit-ut/vc10/syskit-ut
353  user: ttp
354  */
355  DumpFile dumpFile(dumpPath_);
356  Process myProc(Process::myId());
357  wchar_t curDir[MAX_PATH_LENGTH + 1];
358  wchar_t* user = myProc.user();
359  fwprintf(dumpFile, HDR_FORMAT L"%7s: %s\r\n" L"%7s: %s\r\n" L"%7s: %s\r\n",
360  timestamp, myProc.id(), Thread::myId(),
361  L"cmdLine", GetCommandLineW(),
362  L"curDir", getCurDir(curDir),
363  L"user", user);
364  delete[] user;
365 
366  // Show call stack.
367  wchar_t* stack = describe();
368  fwprintf(dumpFile, L"%s", stack);
369  delete[] stack;
370 
371  // Show module addresses.
372  FILE* arg = dumpFile;
373  if (showAddrs)
374  {
375  fwprintf(dumpFile, L"\r\n## addr size basename\r\n");
376  myProc.apply(showAddr, arg);
377  }
378 
379  // Show versions.
380  if (showVersions)
381  {
382  const wchar_t* osVer = Process::osVer();
383  fwprintf(dumpFile, L"\r\n## versions\r\n" L"%7s: %s\r\n", L"osVer", osVer);
384  delete[] osVer;
385  myProc.apply(showModule, arg);
386  }

437 //!
438 //! Describe call stack as a string.
439 //! The returned string is to be freed by caller using the delete[] operator when done.
440 //!
441 wchar_t* CallStack::describe() const
442 {
443  wchar_t hdr[255 + 1];
444  size_t bufSizeInChars = sizeof(hdr) / sizeof(hdr[0]);
445  size_t hdrLength = swprintf(hdr, bufSizeInChars, L"\r\n## call stack (exceptionCode=%08x)\r\n", exceptionCode_);
446 
447  // Compute required string length for all frames.
448  size_t n = hdrLength;
449  for (const Frame* frame = curFrame_; frame != 0; frame = frame->next())
450  {
451  const wchar_t* w = frame->describe();
452  n += wcslen(w);
453  }
454  ++n;
455 
456  // Describe call stack. Example:
457  /*
458  ## call stack (exceptionCode=00000000)
459  00411504: syskit-ut-d!MiscSuite::testCallStack00#188
460  c:/p4c/ipsprobe/syskit-ut/miscsuite.cpp+37#37
461  004230BC: syskit-ut-d!CppUnit::TestCaller<MiscSuite>::runTest#19
462  c:/p4c/ipsprobe/include/cppunit/testcaller.h+173#0
463  :
464  */
465  wchar_t* desc = new wchar_t[n];
466  wchar_t* w = desc;
467  memcpy(w, hdr, hdrLength * sizeof(desc[0]));
468  w += hdrLength;
469  for (const Frame* frame = curFrame_; frame != 0; frame = frame->next())
470  {
471  const wchar_t* p = frame->describe();
472  n = wcslen(p);
473  memcpy(w, p, n * sizeof(w[0]));
474  w += n;
475  }
476  *w = 0;
477 
478  return desc;
479 }
