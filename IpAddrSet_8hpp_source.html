14 //! set of IPv4 addresses
15 class IpAddrSet: protected appkit::U32Set
16  //!
17  //! A class representing a set of IP addresses. Implemented as an
18  //! array of contiguous ranges using U32Set. An IpAddrSet instance
19  //! consists of zero or more IP addresses. Strings can be used to
20  //! specify IP addresses in the set. A valid string specification
21  //! is a sequence of tokens, and each token is either a valid dotted
22  //! IP address or a valid dotted IP address range ("lo-hi"). The token
23  //! delimiter is a comma by default but can be overridden. White-spaces
24  //! surrounding the token delimiter are allowed. Unsigned integers can
25  //! also be used to specify IP addresses in the set. For example,
26  //! 0xac101001U is equivalent to "172.16.16.1". Example:
27  //!\code
28  //! IpAddrSet set;
29  //! set.add("172.16.16.2-172.16.16.11,172.16.16.80");
30  //! set.rm("172.16.16.4");
31  //! set.rm(0xac101006, 0xac101008); //"172.16.16.6-172.16.16.8"
32  //!\endcode
33  //!
