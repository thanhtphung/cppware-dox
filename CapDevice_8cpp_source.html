294  for (unsigned int i = 0; i < numAddrs; ++i)
295  {
296 
297  // Local? (home address of one of the capturing devices)
298  if (addr == device->addr_[i])
299  {

301  }
302 
303  // Broadcast? (broadcast address for a connected subnet)
304  unsigned int netMask = device->netMask_[i];

306  unsigned int broadcast1 = broadcast0 | ~netMask;

310  }
311 
312  // Near? (part of a connected subnet)
313  // Only if it's not a local address nor a broadcast address.
314  if ((category != Near) && (addr & netMask) == broadcast0)
315  {
316  category = Near;
317  }
318  }
319  }

378  {
379  CapDevice* p = curDevice[i];
380  CapDevice* found = findByMac(device_, numDevices_, p->macAddr_);
381  if (found == 0)
382  {

384  {

386  ++numDevices_;

392  {
393  ++numChanges;
394  }
395  }
396

398  return numChanges;

403  unsigned int addr[MaxAddrIndex + 1],
404  unsigned int mask[MaxAddrIndex + 1])
405 {
406  unsigned int numAddrs = 0;
407  for (const pcap_addr_t* p = pcapAddr; p != 0; p = p->next)
408  {
409  if (p->addr->sa_family == AF_INET)
410  {
411  const struct sockaddr_in* socAddr = reinterpret_cast<const struct sockaddr_in*>(p->addr);
412  addr[numAddrs] = ntohl(socAddr->sin_addr.s_addr);
413  socAddr = reinterpret_cast<const struct sockaddr_in*>(p->netmask);
414  mask[numAddrs] = (socAddr != 0)? ntohl(socAddr->sin_addr.s_addr): addr[numAddrs];
415  if (numAddrs++ == CapDevice::MaxAddrIndex)
416  {
417  break;
418  }
419  }
420  }
421 
422  return numAddrs;
