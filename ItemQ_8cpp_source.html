40 //!
41 //! Peek at queue by applying callback to all queued items. The callback should
42 //! return true to continue iterating and should return false to abort iterating.
43 //! Return false if the callback aborted the iterating. Return true otherwise.
44 //! The queue is frozen during this expectedly quick operation, and any attempt
45 //! to modify this queue in the callback will cause deadlocks.
46 //!
47 bool ItemQ::apply(cb0_t cb, void* arg) const
48 {
49  bool ok = true;
50  SpinSection::Lock lock(ss_);
51  for (size_t i = 0, length = numItems(); i < length; ++i)
52  {
53  Item* item = static_cast<Item*>(peek(i));
54  if (!cb(arg, item, i))
55  {
56  ok = false;
57  break;
58  }
59  }
60 
61  return ok;
62 }

146 //
147 // Grow if growable. Return true if grown.
148 //
149 bool ItemQ::grow()
150 {
151  bool grown = false;
152  if (canGrow())
153  {
154 
155  // Limit growth due to semaphore's limit.
156  unsigned int oldCap = capacity();
157  unsigned int newCap = nextCap();
158  if (newCap > MAX_CAP)
159  {
160  newCap = MAX_CAP;
161  }
162 
163  // Reflect growth in semaphore.
164  unsigned int delta = newCap - oldCap;
165  if ((delta > 0) && resize(newCap))
166  {
167  emptySlotCount_->incrementBy(delta);
168  grown = true;
169  }
170  }
171 
172  return grown;
173 }
