532 //! are optional. Use a non-zero resulting dictionary to retrieve the specific difference. The

534 //! another iteration. So if both adds and updates are needed, they should be computed at the

540  {
541 
542  // Iterate this dictionary.
543  // The entries found in the before snapshot have been updated if their values differ.
544  // The entries not in the before snapshot have been added.
545  added->reset();
546  if (updated)

642 {

644  String k;
645  String v;
646  for (txt.reset(); txt.next(line, trimLines);)
647  {
648  size_t startAt = 0;
649  size_t kLength = line.find(kvDelim, startAt);
650  if (kLength != String::INVALID_INDEX)

653  k.trimSpace();
654  v = String(line, kLength + 1, line.length() - kLength - 1);
655  v.trimSpace();

658  }
659  else
