48 bool Paddr::getAddrInfo(unsigned int& addr, unsigned short& port, const char* s, char delim, bool portIsOptional)
49 {
50 
51  // With the dotted notation, if delim is zero, any reasonable delimiter is acceptable.
52  // With a hostname, the delimiter needs to be more specific and the default colon seems
53  // like the best fit to avoid ambiguity. For example, both "xxx" and "xxx-1" can be
54  // valid hostnames.
55  if (delim == 0)
56  {
57  delim = ':';
58  }
59 
60  bool ok;
61  unsigned int addr32;
62  unsigned int port32;
63  const char* delimP = strrchr(s, delim);
64  if (delimP == 0)
65  {
66  if (!portIsOptional)
67  {
68  ok = false;
69  return ok; //no delimiter, but a port number is required
70  }
71  port32 = 0;
72  ok = Winsock::getAddrInfo(addr32, s);
73  }
74  else
75  {
76  const char* p0 = delimP + 1;
77  char* p1;
78  port32 = strtoul(p0, &p1, 0);
79  if ((!S32::isDigit(*p0)) || (p1 == p0) || (*p1 != 0) || (port32 > 65535UL))
80  {
81  ok = false;
82  return ok; //invalid port number
83  }
84  ok = Winsock::getAddrInfo(addr32, s, delimP - s);
85  }
86 
87  if (ok)
88  {
89  addr = addr32;
90  port = static_cast<unsigned short>(port32);
91  }
92 
93  return ok;
94 }

97 //!
98 //! Reset instance with given port address (e.g., "172.16.16.1:8080"). If
99 //! portIsOptional is set, then assume port is zero (e.g., "172.16.16.1"
100 //! would be the same as "172.16.16.1:0"). If delim is non-zero, given port
101 //! address must use that delimiter to delimit the address and the port parts.
102 //! If delim is zero, any delimiter is acceptable. Return true if s specifies
103 //! a valid port address. Use of a hostname instead of the dotted notation is
104 //! allowed. See IpAddr for details.
105 //!
106 bool Paddr::reset(const char* s, char delim, bool portIsOptional)
107 {
108 
109  // Sanity check.
110  if (s == 0)
111  {
112  bool ok = false;
113  return ok;
114  }
115 
116  // String s must start with "a.b.c." (e.g., "172.16.16.").
117  char* p;
118  unsigned int abcde[5];
119  for (size_t i = 0; i < 3; ++i, s = p + 1)
120  {
121  abcde[i] = strtoul(s, &p, 0);
122  if ((!S32::isDigit(*s)) || (p == s) || (*p != DOT) || (abcde[i] > 255UL))
123  {
124  bool ok = (i == 0) && getAddrInfo(u32_[Addr], u16_[Port], s, delim, portIsOptional);
125  return ok;
126  }
127  }
128 
129  // String s must start with "a.b.c.d" (e.g., "172.16.16.1").
130  abcde[3] = strtoul(s, &p, 0);
131  if ((!S32::isDigit(*s)) || (p == s) || (abcde[3] > 255UL))
132  {
133  bool ok = false;
134  return ok;
135  }
136 
137  bool delimIsValid = (delim == 0)? (S32::isPunct(*p)): (*p == delim);
138  if (delimIsValid)
139  {
140  s = p + 1;
141  abcde[4] = strtoul(s, &p, 0);
142  if ((!S32::isDigit(*s)) || (p == s) || (*p != 0) || (abcde[4] > 65535UL))
143  {
144  bool ok = false;
145  return ok;
146  }
147  }
148  else
149  {
150  if ((!portIsOptional) || (*p != 0))
151  {
152  bool ok = false;
153  return ok;
154  }
155  abcde[4] = 0;
156  }
157 
158  // Return true to indicate s specifies a valid port address.
159  u32_[Addr] = (abcde[0] << 24) | (abcde[1] << 16) | (abcde[2] << 8) | abcde[3];
160  u16_[Port] = static_cast<unsigned short>(abcde[4]);
161  bool ok = true;
162  return ok;
163 }

168 //! if *item0 < *item1. Return 0 if *item0 == *item1. Return a positive value if

172 {

175 
176  int rc;

178  {
179  rc = -1;
180  }
181  else if (k0.u32_[Addr] > k1.u32_[Addr])
182  {
183  rc = 1;
184  }
185  else if (k0.u16_[Port] < k1.u16_[Port])
186  {
187  rc = -1;
188  }
189  else

192  }
