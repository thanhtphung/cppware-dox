74 {

76  socAddr.sin_family = AF_INET;
77  socAddr.sin_addr.s_addr = htonl(addr);
78  socAddr.sin_port = htons(port);

84 //! Get socket binding. Return true if successful.
85 //!
86 bool UdpClient::getMyAddr(Paddr& paddr) const
87 {
88  bool ok;
89  struct sockaddr_in socAddr;
90  socklen_t socAddrLen = sizeof(socAddr);
91  if ((soc_ != INVALID_SOCKET) &&
92  (getsockname(soc_, (struct sockaddr*)(&socAddr), &socAddrLen) == 0) &&
93  (socAddrLen == sizeof(socAddr)) &&
94  (socAddr.sin_family == AF_INET))
95  {
96  paddr.setAddr(ntohl(socAddr.sin_addr.s_addr));
97  paddr.setPort(ntohs(socAddr.sin_port));
98  ok = true;
99  }
100  else
101  {
102  ok = false;
103  }
104 
105  return ok;
106 }
