97 //!
98 //! Just as a string can be expanded into a vector of strings, a vector of strings
99 //! can be collapsed back into one string. This method collapses the vector into a
100 //! string by concatenating up to maxItems members. The members are separated by
101 //! delim if delim is non-zero. A trailing "..." is used to indicate there are more
102 //! members in the vector. Return the resulting concatenated string.
103 //!
104 String StringVec::stringify(const char* delim, size_t maxItems) const
105 {
106  String s;
107  const String* const* p0 = raw();
108  size_t length = (numItems() <= maxItems)? numItems(): maxItems;
109  for (size_t i = 0; i < length; ++i)
110  {
111  if ((delim != 0) && (i > 0))
112  {
113  s += delim;
114  }
115  s += *p0[i];
116  }
117 
118  if (length < numItems())
119  {
120  s += "...";
121  }
122 
123  return s;
124 }
