18 //! set of unsigned 16-bit numbers
19 class U16Set: public Set
20  //!
21  //! A class representing a set of unsigned short keys. Implemented
22  //! as an array of contiguous ranges. A U16Set instance has type
23  //! "U16" if type() is used. A U16Set instance consists of zero
24  //! or more valid unsigned short keys. Valid unsigned short keys
25  //! have range [validMin()..validMax()]. Strings can be used
26  //! to specify keys in a set. A valid string specification is a
27  //! sequence of tokens, and each token is either a valid key or a
28  //! valid key range ("lo-hi"). The token delimiter is a comma by
29  //! default but can be overridden. White-spaces surrounding the
30  //! token delimiter are allowed. Example:
31  //!\code
32  //! U16Set set;
33  //! set.add(2, 11); //keys: 2-11
34  //! set.rm(4); //keys: 2-3,5-11
35  //! set.rm("6,8-10"); //keys: 2-3,5,7,11
36  //!\endcode
37  //!
