46  if (this == &q)
47  {

51  // Might need to grow to accomodate source.
52  unsigned int minCap = q.numItems();
53  if (minCap > capacity())
54  {
55  if (canGrow())
56  {
57  reset();
58  resize(nextCap(minCap));
59  }
60  }
