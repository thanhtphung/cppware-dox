56 //! Quick implementation for the case-insensitive strstr().
57 //!
58 const char* strcasestr(const char* haystack, const char* needle)
59 {
60  const unsigned char* hp = 0;
61  const unsigned char* np = 0;
62  for (const unsigned char* p = reinterpret_cast<const unsigned char*>(haystack); *p != 0; ++p)
63  {
64  if (np != 0)
65  {
66  if (s_icCharMap[*p] == s_icCharMap[*np])
67  {
68  if (*++np == 0)
69  {
70  return reinterpret_cast<const char*>(hp);
71  }
72  }
73  else
74  {
75  np = 0;
76  }
77  }
78  else if (s_icCharMap[*p] == s_icCharMap[*reinterpret_cast<const unsigned char*>(needle)])
79  {
80  np = reinterpret_cast<const unsigned char*>(needle)+1;
81  hp = p;
82  }
83  }
84 
85  return 0;
86 }

94 String& Str::stripSpace(String& result, const char* s, char delim)
95 {
96 
97  // Look for given delimiter or hyphen, then strip consecutive
98  // white spaces surrounding the located character.
99  result.reset();
100  for (const char* s1 = s; *s1;)
101  {
102  if ((*s1 == delim) || (*s1 == '-'))
103  {
104  unsigned int numSpaces = 0;
105  for (const char* p = s1 - 1; p >= s; --p)
106  {
107  if (S32::isSpace(*p))
108  {
109  ++numSpaces;
110  }
111  else
112  {
113  break;
114  }
115  }
116  if (numSpaces > 0)
117  {
118  result.truncate(result.length() - numSpaces);
119  }
120  result.append(1, *s1);

122  }
123  else

125  result.append(1, *s1++);
126  }
127  }
