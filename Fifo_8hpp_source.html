14 //! first-in-first-out queue of opaque items
15 class Fifo: public Growable
16  //!
17  //! A class representing a FIFO queue of opaque items. The queue has
18  //! an initial capacity of capacity() items. The capacity can grow
19  //! as needed if the growth factor is set to non-zero when constructed
20  //! or afterwards using setGrowth(). If the queue is growable, growth
21  //! can occur when items are added non-circularly. For a FIFO queue,
22  //! items are normally added at the tail and removed from the head. As
23  //! exceptions, items can be added at the head and can also be removed
24  //! from the tail. Implemented using a contiguous array. Example:
25  //!\code
26  //! Fifo q;
27  //! q.add(item0);
28  //! :
29  //! void* item1;
30  //! q.rm(item1);

108 //! Peek at given index and return the residing item. Don't do any error
109 //! checking. Behavior is unpredictable if given index is invalid. This
110 //! provides easy access to all items in the queue. The item at index zero
111 //! corresponds to the item at the head of the queue, and the item at index
112 //! numItems()-1 corresponds to the item at the tail of the queue.
113 inline Fifo::item_t Fifo::peek(size_t index) const
114 {
115  size_t i = head_ + index;
116  return (i < capacity())? item_[i]: item_[i - capacity()];
117 }

119 //! Add given item to the head of the queue in a circular manner.
120 //! If the queue is full, the item at the tail is removed from the
121 //! queue before the given item is added. Return true if successful.
122 //! Return false otherwise (queue has a zero capacity).
123 inline bool Fifo::addAtHeadCircularly(item_t item)
124 {
125  item_t droppedItem;
126  if (numItems_ >= capacity()) rmFromTail(droppedItem);
127  return addAtHead(item);
128 }
