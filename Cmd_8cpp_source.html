113 //!
114 //! Form and send given null-terminated response. Assume response is
115 //! partial if not double-null-terminated and complete otherwise. For
116 //! examples, formRsp(req, "partial") indicates a partial response and
117 //! formRsp(req, "complete%c", 0) indicates a complete response. Return
118 //! true if successful.
119 //!
120 bool Cmd::formRsp(const CmdLine& req, const char* format, ...)
121 {
122  bool ok;
123  const tag_t& tag = *static_cast<const tag_t*>(req.tag());
124  if (tag.rspPort == (tag.ioPort->asWord() & 0x0000ffffUL))
125  {
126  va_list marker;
127  va_start(marker, format);
128  char rsp[MaxRspLength + 1];
129  int rspLength = vsnprintf(rsp, sizeof(rsp), format, marker);
130  va_end(marker);
131  ok = ((rspLength > 0) && (rspLength < (int)(sizeof(rsp))))?
132  tag.cmd->sndRsp(req, rsp, rspLength):
133  false;
134  }
135  else
136  {
137  ok = false;
138  }
139 
140  return ok;
141 }

194 bool Cmd::sndRsp(const CmdLine& req, const void* rsp, size_t rspLength)
195 {
196  const tag_t& tag = *static_cast<const tag_t*>(req.tag());
197  UdpClient* io = tag.io;
198  Paddr src(tag.rspAddr, tag.rspPort);
199  size_t countIndex = tag.cmdIndex * CountsPerName;
200 
201  // Allow empty responses.
202  // Packetize long responses.
203  bool ok;
204  const unsigned char* pkt = static_cast<const unsigned char*>(rsp);
205  size_t more = rspLength;
206  unsigned int pktCount = 0;
207  unsigned int timeoutInMsecs = SND_TIMEOUT;
208  do
209  {
210  size_t pktLength = (more > MAX_PKT_LENGTH)? MAX_PKT_LENGTH: more;
211  ok = io->snd(pkt, pktLength, src, timeoutInMsecs);
212  ++pktCount;
213  pkt += pktLength;
214  more -= pktLength;
215  } while ((more > 0) && ok);
216 
217  // Collect some stats.
218  if (ok)
219  {
220  count_[countIndex + RspPkts] += pktCount;
221  size_t count64Index = tag.cmdIndex * Count64sPerName;
222  count64_[count64Index + RspBytes] += rspLength;
223  }
224  else
225  {
226  ++count_[countIndex + RspFails];
227  }
228 
229  // Give application an opportunity to look at the response.
230  SysIo* sysIo = tag.sysIo;
231  sysIo->onRsp(req, static_cast<const char*>(rsp), rspLength, ok);
232  return ok;
233 }

246 //!
247 //! A command is typically run using a few components in the framework including
248 //! CmdAgent (command client) and SysIo (command server). That is sometimes not
249 //! feasible. This method offers a way to run (i.e., sprint) commands w/o those
250 //! components. With this method, the onRun() method is invoked by the caller thread
251 //! (vs. a dedicated thread), and there's no client/server communications. Return
252 //! true if successful.
253 //!
254 bool Cmd::sprint(const CmdLine& req, String& rsp)
255 {
256  bool ok = false;
257 
258  PrimitiveIo sysIo;
259  const String* cmdName = req.arg(0);
260  if (cmdName != 0)
261  {
262  UdpClient* io = UdpClient::createNull();
263  tag_t cmdTag(&sysIo, io, stdout, sysIo.ioPort(0));
264  cmdTag.cmd = map_->find(cmdName->ascii(), cmdTag.cmdIndex);
265  if (cmdTag.cmd == this)
266  {
267  void* oldTag = req.tag();
268  req.setTag(&cmdTag);
269  size_t countIndex = cmdTag.cmdIndex * CountsPerName;
270  ++count_[countIndex + Sprints];
271  ok = onRun(req); //allow commands to be self-destructive at onRun() -- avoid further Cmd usage
272  req.setTag(oldTag);
273  }
274 
275  delete io;
276  }
277 
278  rsp = sysIo.rsp();
279  return ok;
280 }
