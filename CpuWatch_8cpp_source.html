38 //!
39 //! Compute the CPU usage. This method is valid only after a sequence
40 //! of start()-stop() invocations. It returns zeroes if the watch was
41 //! not properly stopped. Upon return, <i>sysUsage</i> holds the
42 //! system CPU time as a percentage of the elapsed time, <i>usrUsage</i>
43 //! holds the user CPU time as a percentage of the elapsed time. The
44 //! elapsed time is the stop-start time difference.
45 //!
46 void CpuWatch::computeUsage(double& sysUsage, double& usrUsage) const
47 {
48  if (delta_)
49  {
50  sysUsage = static_cast<double>(sysTime_);
51  sysUsage /= static_cast<double>(delta_);
52  sysUsage *= 100.0;
53  usrUsage = static_cast<double>(usrTime_);
54  usrUsage /= static_cast<double>(delta_);
55  usrUsage *= 100.0;
56  }
57  else
58  {
59  sysUsage = 0.0;
60  usrUsage = 0.0;
61  }
62 }

103 //!
104 //! Stop the CPU watch. Record the stop time and compute
105 //! the stop-start time difference. Also collect the CPU
106 //! time used in the stop-start duration. This operation
107 //! does not reset the start time. A subsequent consecutive
108 //! stop would overwrite the earlier stop time.
109 //!
110 void CpuWatch::stop()
111 {
112  stopTime_ = TickTime::curTime();
113  (this->*getCpuTime_)();
114  sysTime_ -= sysTime0_;
115  usrTime_ -= usrTime0_;
116  cpuTimeToTicks(sysTime_, usrTime_);
117  delta_ = stopTime_ - startTime_;
118 }
