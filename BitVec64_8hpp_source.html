15 class BitVec64
16  //!
17  //! A class representing a vector of bits. Most operations are trivial (e.g.,
18  //! clear(), set(), reset(), etc.). Implemented using an array of 64-bit words. Bit 0
19  //! is the least significant bit of the first word, bit BitsPerWord-1 is the
20  //! most significant bit of the first word, and bit BitsPerWord is the least
21  //! significant bit of the second word, etc. As a convenience, BitVec is an
22  //! alias of BitVec64 in x64 builds. Example:
23  //!\code
24  //! BitVec64 vec(256, false); //start with 256 clear bits
25  //! vec.set(100); //set bit 100
26  //! vec.set(9, 88); //set bits 9 through 88
27  //! vec.clearAll(); //clear all bits
28  //! vec.invert(); //invert all bits

170 // Given the last word of a raw bit vector, clear the unused bits
171 // and return the result. This raw bit vector has maxBits bits.
172 inline BitVec64::word_t BitVec64::clearUnusedBits(word_t word, size_t maxBits)
173 {
174  return ((maxBits & 0x3fU) != 0)? (word & ((1ULL << (maxBits & 0x3fU)) - 1)): word;
175 }

177 // Given the last word of a raw bit vector, set the unused bits
178 // and return the result. This raw bit vector has maxBits bits.
179 inline BitVec64::word_t BitVec64::setUnusedBits(word_t word, size_t maxBits)
180 {
181  return ((maxBits & 0x3fU) != 0)? (word | ~((1ULL << (maxBits & 0x3fU)) - 1)): word;
182 }

197 //! Return true/false if given bit is set/clear. Don't do any error checking.
198 inline bool BitVec64::operator [](size_t bit) const
199 {
200  size_t vecI = bit >> 6;
201  size_t bitI = bit & 0x3fU;
202  return ((raw_[vecI] & (1ULL << bitI)) != 0);
203 }
