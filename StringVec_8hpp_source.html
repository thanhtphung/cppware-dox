17 //! vector of strings
18 class StringVec: private syskit::Vec
19  //!
20  //! A class representing an unsorted vector of strings. The vector has
21  //! an initial capacity of capacity() items. The capacity can grow as
22  //! needed if the growth factor is set to non-zero when constructed or
23  //! afterwards using setGrowth(). If the vector is growable, growth can
24  //! occur when items are added. Items are added using various forms of
25  //! the add() method and removed using various forms of the rm() method.
26  //! Updates can be expensive if many items need to be shifted up or down.
27  //! For best update performance, items should be added to the tail end,
28  //! and items should be removed without maintaining the current ordering.
29  //! Example:
30  //!\code
31  //! StringVec vec;
32  //! vec.add(item0);
33  //! vec.addAtIndex(1 /*index*/, item1);
34  //! vec.rm(item2, false /*maintainOrder*/);
35  //! vec.rm(item3, true /*maintainOrder*/);
36  //! vec.rmFromIndex(1 /*index*/, false /*maintainOrder*/);
37  //! vec.rmFromIndex(0 /*index*/, true /*maintainOrder*/);
38  //!\endcode
39  //!
