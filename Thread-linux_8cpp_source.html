22 static void* start(void* arg)
23 {
24 
25  // Utilize entry point and argument specified by creating user.
26  bool crashed;
27  void* rc;
28  Thread::Entry* entry = static_cast<Thread::Entry*>(arg);
29  try
30  {
31  rc = entry->enter();
32  crashed = false;
33  }
34  catch (...)
35  {
36  rc = static_cast<unsigned char*>(0) - 1;
37  crashed = true;
38  }
39 
40  // Notify monitor of crash.
41  if (crashed)

43  void* arg;
44  Thread::crashCb_t crashCb = Thread::crashCb(arg);
45  crashCb(arg);
46  }
47 
48  // Return the thread exit code.
49  delete entry;
50  return rc;

54 //!
55 //! Create and start a thread with given characteristics. That is, the
56 //! new thread has stack size of stackSizeInBytes bytes. Its entry point
57 //! is entrance. When the entry point is entered, the argument arg is
58 //! used as an opaque parameter. Start thread in suspended (vs. running)
59 //! mode if startSuspended is true. Suspended threads remain idle until
60 //! resumed via resume(). The construction can fail due to lack of system
61 //! resources. Use isOk() to determine if the thread construction is
62 //! successful.
63 //!
64 Thread::Thread(entrance_t entrance, void* arg, unsigned int stackSizeInBytes, bool /*startSuspended*/)
65 {
66  bool ok;
67  pthread_attr_t attr = PTHREAD_ATTR0;
68  pthread_attr_t* attrP = 0;
69  if (stackSizeInBytes)
70  {
71  attrP = &attr;
72  pthread_attr_init(attrP);
73  ok = (pthread_attr_setstacksize(attrP, stackSizeInBytes) == 0);
74  }
75  else
76  {
77  ok = true;
78  }
79 
80  state_ = new State();
81  if (ok)
82  {
83  Entry* entry = new Entry(entrance, arg, state_);
84  if (pthread_create(&id_, attrP, start, entry) == 0)
85  {
86  thread_ = this;
87  return;
88  }
89  delete entry;
90  }
91 
92  id_ = INVALID_ID;
93  thread_ = 0;
94 }

97 //!
98 //! Suspend calling thread's execution for at least napTimeInMsecs msecs
99 //! unless interrupted. Return true if the nap completes successfully.
100 //! Return false otherwise (the nap was interrupted).
101 //!
102 bool Thread::takeANap(unsigned int napTimeInMsecs, bool /*beAlert*/)
103 {
104  struct timespec napTime;
105  const unsigned int MSECS_PER_SEC = 1000U;
106  const unsigned int NANOSECS_PER_MSEC = 1000000U;
107  napTime.tv_sec = napTimeInMsecs / MSECS_PER_SEC;
108  napTime.tv_nsec = (napTimeInMsecs % MSECS_PER_SEC) * NANOSECS_PER_MSEC;
109  bool ok = (nanosleep(&napTime, 0) == 0);
110  return ok;
111 }
