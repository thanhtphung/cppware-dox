54 // Return true if given UTF8 multi-byte sequence (up to MaxSeqLength bytes
55 // starting at seq) is valid. A valid UTF8 byte sequence has between
56 // MinSeqLength and MaxSeqLength bytes, represents a value smaller
57 // than or equal to MaxValue, and is as short as possible.
58 //
59 bool Utf8::isValid234(const utf8_t seq[MaxSeqLength])
60 {
61 
62  // Basically, subsequent bytes must have range 0x80U..0xbfU. If
63  // sequence length is 2, range check is not required since it's
64  // already built-in in the sequence length table. If sequence
65  // length is more than 2, the sequence is valid only if its value
66  // cannot be encoded using fewer bytes. A 1-byte sequence is an
67  // ASCII character. Also, check for reserved values in 3-byte
68  // sequences and out-of-range values in 4-byte sequences.
69  switch (seqLength_[seq[0]])
70  {
71  case 4:
72  if ((seq[3] < 0x80U) || (seq[3] > 0xbfU))
73  {
74  return false;
75  }
76  // no break
77 
78  case 3:
79  if ((seq[2] < 0x80U) || (seq[2] > 0xbfU))
80  {
81  return false;
82  }
83  // no break
84 
85  case 2:
86  if (seq[1] > 0xbfU)
87  {
88  return false;
89  }
90  if (seq[0] == 0xe0U) return (seq[1] >= 0xa0U); //too many bytes in a 3-byte sequence?
91  if (seq[0] == 0xedU) return (seq[1] <= 0x9fU); //reserved value in a 3-byte sequence?
92  if (seq[0] == 0xf0U) return (seq[1] >= 0x90U); //too many bytes in a 4-byte sequence?
93  if (seq[0] == 0xf4U) return (seq[1] <= 0x8fU); //value too big for a 4-byte sequence?
94  return (seq[1] >= 0x80U);
95 
96  default:
97  return false;
98  }
99 }

159 // Decode given UTF8 multi-byte sequence (up to MaxSeqLength bytes starting at
160 // seq). Assume the sequence has been validated, so minimize error checking.
161 // If successful, update instance with decoded value and return the number
162 // of bytes decoded (2, 3, or 4). Otherwise (i.e., given UTF8 byte sequence
163 // is malformed), return zero.
164 //
165 unsigned int Utf8::decode234(const utf8_t seq[MaxSeqLength])
166 {
167 
168  // Each subsequent byte holds 6 bits. The first byte holds up
169  // to 5 high-order bits (seqLength=n --> numHighOrderBits=7-n).
170  unsigned int seqLength = seqLength_[seq[0]];
171  unsigned int value = 0;
172  unsigned int delta = 0;
173  switch (seqLength)
174  {
175  case 4:
176  value = seq[3] & 0x3fU;
177  delta = 6;
178  // no break
179 
180  case 3:
181  value |= (seq[2] & 0x3fU) << delta;
182  delta += 6;
183  // no break
184 
185  case 2:
186  value |= (seq[1] & 0x3fU) << delta;
187  delta += 6;
188  value |= (seq[0] & FIRST_BYTE_D_MASK[seqLength]) << delta;
189  c_ = value;
190  break;
191 
192  default:
193  seqLength = 0;
194  break;
195  }
196 
197  return seqLength;
198 }

272 // Encode non-ASCII value into a UTF8 byte sequence. Save results in seq.
273 // Assume seq is a buffer large enough for this value. Return the resulting
274 // sequence length (2, 3, or 4).
275 //
276 unsigned int Utf8::encode234(utf8_t seq[MaxSeqLength]) const
277 {
278 
279  // Determine sequence length for a multi-byte sequence.
280  unsigned int seqLength;
281  if (c_ <= MAX_VALUE_2) seqLength = 2;
282  else if (c_ <= MAX_VALUE_3) seqLength = 3;
283  else seqLength = 4;
284 
285  // Each subsequent byte holds 6 bits. The first byte holds up
286  // to 5 high-order bits (seqLength=n --> numHighOrderBits=7-n).
287  unsigned int value = c_;
288  switch (seqLength)
289  {
290  case 4:
291  seq[3] = static_cast<unsigned char>((value & 0x3fU) | 0x80U);
292  value >>= 6;
293  // no break
294 
295  case 3:
296  seq[2] = static_cast<unsigned char>((value & 0x3fU) | 0x80U);
297  value >>= 6;
298  // no break
299 
300  default:
301  seq[1] = static_cast<unsigned char>((value & 0x3fU) | 0x80U);
302  seq[0] = static_cast<unsigned char>((value >> 6) | FIRST_BYTE_E_MASK[seqLength]);
303  break;
304 
305  }
306 
307  return seqLength;
