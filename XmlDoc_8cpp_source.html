104 //!
105 //! Locate descendent given the full name (e.g., "/grandma/mom/me"). Return
106 //! located descendent. Return zero if not found.
107 //!
108 XmlElement* XmlDoc::findByName(const String& fullName) const
109 {
110 
111  // Not a full name?
112  const char* line;
113  size_t length;
114  DelimitedTxt txt(fullName, false /*makeCopy*/, XmlElement::NAME_DELIM);
115  if (!txt.next(line, length))
116  {
117  return 0;
118  }
119 
120  // Mismatched root?
121  String name;
122  if ((!txt.next(name)) || (txt.trimLine(name) != root_->name()))
123  {
124  return 0;
125  }
126 
127  // Find descendent. Start from root. One generation at a time.
128  XmlElement* found = root_;
129  while (txt.next(name))
130  {
131  txt.trimLine(name);
132  found = found->findKid(name);
133  if (found == 0)
134  {
135  break;
136  }
137  }
138 
139  return found;
140 }

198 void XmlDoc::construct(const MappedTxtFile& file)
199 {
200  if (file.isOk())
201  {
202  XmlDoc* doc;
203  const Bom& bom = file.bom();
204  if ((bom == Bom::None) || (bom == Bom::Utf8))
205  {
206  XmlLexer lexer(file.image(), static_cast<size_t>(file.size()));
207  doc = lexer.scan();
208  }
209  else
210  {
211  String lexee;
212  lexee.resetX(bom, file.image(), static_cast<size_t>(file.size()));
213  XmlLexer lexer(lexee);
214  doc = lexer.scan();
215  }
216  errDesc_ = doc->errDesc_;
217  prolog_ = doc->prolog_;
218  root_ = doc->root_;
219  doc->prolog_ = 0;
220  doc->root_ = 0;
221  delete doc;
222  }
223  else
224  {
225  errDesc_ = "cannot access file";
226  prolog_ = new XmlProlog;
227  root_ = new UnknownElement(String());
228  }
229 }
