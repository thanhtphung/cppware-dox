81 //! If it cannot grow, smaller items from the source heap will be dropped.
82 //!
83 const U32Heap& U32Heap::operator =(const U32Heap& heap)

86  // Prevent self assignment.
87  if (this == &heap)
88  {
89  return *this;
90  }
91

93  // Might need to truncate if not growable.
94  unsigned int curCap = capacity();
95  unsigned int minCap = heap.numItems_;
96  if (minCap > curCap)
97  {
98  if (canGrow())
99  {
100  delete[](item_ + 1);
101  item_ = new item_t[setNextCap(minCap)];
102  --item_;
103  }
104  else
105  {
106  minCap = curCap; //truncate copy
107  }
108  }
109 
110  // Don't do much if source heap is empty or if this heap has zero capacity.
111  if (minCap == 0)
112  {
113  numItems_ = 0;
114  }
115 
116  // Copy all items by copying memory if the comparison functions are
117  // identical and if no truncation is required.
118  else if (minCap == heap.numItems_)
119  {
120  numItems_ = minCap;
121  memcpy(item_ + 1, heap.item_ + 1, numItems_ * sizeof(*item_));
122  }
123

125  // the smaller items.
126  else

129  reset();

131  {

136  }

258 //
259 // Start at node i, move downward to restore the heap condition.
260 //
261 void U32Heap::heapifyDown(size_t i)
262 {
263  item_t momV;
264  for (size_t mom = i;;)
265  {
266 
267  // Be more careful with the last mom as there might be only one kid.
268  size_t kid = (mom << 1);
269  size_t sib = kid + 1;
270  if (sib > numItems_)
271  {
272  if ((kid <= numItems_) && ((momV = item_[mom]) < item_[kid]))
273  {
274  item_[mom] = item_[kid];
275  item_[kid] = momV;
276  }
277  break;
278  }
279 
280  // Swap mom with the larger kid and keep going.
281  if (item_[kid] < item_[sib]) kid = sib;
282  if ((momV = item_[mom]) < item_[kid])
283  {
284  item_[mom] = item_[kid];
285  item_[kid] = momV;
286  mom = kid;
287  continue;
288  }
289 
290  // Done if both kids are larger than the mom.
291  break;
292  }
293 }

302  {

304  size_t mom = (kid >> 1);
305  if ((mom > 0) && ((momV = item_[mom]) < item_[kid]))
306  {
307  item_[mom] = item_[kid];
308  item_[kid] = momV;
309  kid = mom;
310  continue;
311  }

317 //!
318 //! Sort given items in-place. Order is ascending if reverseOrder
319 //! is false and is descending if reverseOrder is true.
320 //!
321 void U32Heap::sort(item_t* item, size_t itemCount, bool reverseOrder)
322 {
323 
324  // Construct a temporary heap utilizing the same item array. Then remove
325  // all items from the temporary heap and store them back in the same item
326  // array so that it is sorted in ascending order.
327  U32Heap heap(item, itemCount);
328  for (item_t* dst = item + itemCount - 1; dst >= item; --dst)
329  {
330  item_t next = *dst;
331  *dst = heap.item_[1];
332  heap.item_[1] = next;
333  --heap.numItems_;
334  heap.heapifyDown(1);
335  }
336 
337  // If necessary, reverse item array in-place (e.g., [1, 2, 9] --> [9, 2, 1]).
338  if (reverseOrder)
339  {
340  item_t* left = item;
341  item_t* right = left + itemCount - 1;
342  for (; left < right; ++left, --right)
343  {
344  item_t tmp = *left;
345  *left = *right;
346  *right = tmp;
347  }
348  }
349 
350  // No-op the destructor.
351  heap.item_ = static_cast<item_t*>(0) - 1;
352 }
