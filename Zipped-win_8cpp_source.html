75 {

78  wchar_t siblingPath[MAX_PATH_LENGTH + MAX_BASENAME_LENGTH + 1];

86  {

88  }
89

96  if (_wcsicmp(normalized, siblingPath) == 0)

98  ok = zipItem(sibling, basename, compressionLevel, filetime.ft64);

152 bool Zipped::zipItem(const MappedFile& item, const wchar_t* basename, int compressionLevel, unsigned long long itemFiletime)
153 {
154  char s[MAX_BASENAME_LENGTH * 3 + 1]; //assume worst-case non-ASCII 3 bytes per character
155  WideCharToMultiByte(CP_UTF8, 0, basename, -1, s, sizeof(s), 0, 0);
156  zip_fileinfo fi;
157  const zip_fileinfo* fip = setFileInfo(&fi, itemFiletime);
158  zipOpenNewFileInZip(zip_, s, fip, 0, 0, 0, 0, 0, Z_DEFLATED, compressionLevel);
159 
160  unsigned int numMaps = item.numMaps();
161  unsigned int bytesToWrite = (numMaps > 1)? mapSize_: static_cast<unsigned int>(item.size());
162  unsigned long long remainingBytes = item.size();
163  for (unsigned int i = 0; i < numMaps;)
164  {
165  const unsigned char* src = item.map(i);
166  if (++i == numMaps)
167  {
168  bytesToWrite = static_cast<unsigned int>(remainingBytes);
169  }
170  zipWriteInFileInZip(zip_, src, bytesToWrite);
171  remainingBytes -= bytesToWrite;
172  }
173 
174  zipCloseFileInZip(zip_);
175  bool ok = true;
176  return ok;
177 }
