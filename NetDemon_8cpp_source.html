70 MappedFile* NetDemon::mapRcFile() const
71 {
72 
73  // No problem if no accessible rc3 file. Allow use of same rc3 file for both debug
74  // and release builds (e.g., x.rc3 can be used by both x.exe and xd.exe).
75  String rcPath(confDir());
76  rcPath += Path(path()).base();
77  rcPath += ".rc3";
78  bool readOnly = true;
79  MappedFile* rcFile = new MappedFile(rcPath.widen(), readOnly);
80  if (!rcFile->isOk())
81  {
82  bool ignoreCase = true;
83  if (rcPath.endsWith("d.rc3", ignoreCase))
84  {
85  rcPath.truncate(rcPath.length() - 5);
86  rcPath += ".rc3";
87  rcFile->remap(rcPath.widen(), readOnly);
88  }
89  }
90 
91  return rcFile;
92 }

167 unsigned int NetDemon::countDumps() const
168 {
169  unsigned int numDumps = 0;
170  bool skipNormalization = true;
171  Path myPath(path(), skipNormalization);
172  String prefix(myPath.base());
173  prefix += "-dump-";
174 
175  bool ignoreCase = true;
176  Directory dir(logDir());
177  StringVec children;
178  dir.list(children);
179  for (unsigned int i = 0, numChildren = children.numItems(); i < numChildren; ++i)
180  {
181  if (children.peek(i).startsWith(prefix, ignoreCase))
182  {
183  ++numDumps;
184  }
185  }
186 
187  return numDumps;
188 }

251 //
252 // Run start-up commands from rc3 file, if any.
253 //
254 void NetDemon::runStartUpCmds(const MappedFile& rcFile) const
255 {
256 
257  // Run start-up commands from rc3 file using loopback agent.
258  bool makeCopy = false;
259  DelimitedTxt rcTxt(reinterpret_cast<const char*>(rcFile.map(0UL /*index*/)), static_cast<size_t>(rcFile.size()), makeCopy);
260  String cmd;
261  bool doTrimLine = true;
262  while (rcTxt.next(cmd, doTrimLine))
263  {
264  if ((!cmd.empty()) && (*cmd.ascii() != COMMENT_INDICATOR))
265  {
266  loopbackAgent_->runCmd(cmd);
267  }
268  }
269 }
