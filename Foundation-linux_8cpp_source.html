50  // Use an environment variable to determine the first per-process reference.
51  // Use another environment variable as per-process memory. This is a
52  // relatively simple way to make Foundation a per-process singleton.
53  char tag[20 + 1]; //big enough for "0xnnnnnnnn" or "0xnnnnnnnnnnnnnnnn"
54  unsigned long long startTime = rdtsc();
55  sprintf(tag, "%llu", startTime);
56  setenv(START_TIME, tag, 0 /*overwrite*/);
57  const char* s = getenv(START_TIME);
58  bool isFirstReference = ((s != 0) && (strcmp(s + sizeof(START_TIME), tag) == 0));
59  if (isFirstReference)
60  {
61  unsigned char* p = new unsigned char[FOUNDATION_SIZE];
62  memset(p, 0, FOUNDATION_SIZE);
63  foundation_ = reinterpret_cast<foundation_t*>(p);
64  }
65  else

68  foundation_ = reinterpret_cast<foundation_t*>(strtoull(tag, 0, 0));

112  // Allow each binary to create its own instance.
113  static Foundation* s_foundation = new Foundation;
114 
115  // But only the first instance survives. All subsequent instances are immediately
116  // destroyed upon creation. This code seems unnecessary complicated and inefficient,
117  // and it is. Historically, this object used shared memory and was synchronized with
118  // other processes, and it should be simplified and cleaned up as that feature is
119  // no longer supported.
120  Foundation* foundation = s_foundation->foundation_->outerSkin;
121  if (foundation != s_foundation)
122  {
123  const Foundation* tmp = s_foundation;
124  s_foundation = foundation;
125  delete tmp;
126  }
