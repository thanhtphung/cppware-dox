14 //! condition variable and its associated mutex
15 class CondVar
16  //!
17  //! A class representing a condition variable and its associated mutex.
18  //! A mutex is usually used to prevent other threads from executing some
19  //! code concurrently. A condition variable and its associated mutex is
20  //! a synchronization device that is usually used to wait and let other
21  //! threads run until some condition is met. There are basically two
22  //! categories of CondVar users: an observer and a notifier. An observer
23  //! uses the wait() methods to wait until some condition is met. A notifier
24  //! uses the signal() method to wake up one observer or the broadcast()
25  //! method to wake up all observers. Example:
26  //!\code
27  //! static CondVar s_condVar;
28  //! :
29  //! // observer code --wait until some condition is met
30  //! s_condVar.lockMutex();
31  //! while (/*conditionNotMet*/)
32  //! {
33  //! s_condVar.wait();
34  //! }
35  //! s_condVar.unlockMutex();
36  //! :
37  //! // notifier code --wake up all observers
38  //! s_condVar.lockMutex();
39  //! doSomethingSoConditionIsMet();
40  //! s_condVar.broadcast();
41  //! s_condVar.unlockMutex();
42  //! :
43  //! // notifier code --wake up one observer
44  //! s_condVar.lockMutex();
45  //! doSomethingSoConditionIsMet();
46  //! s_condVar.signal();
47  //! s_condVar.unlockMutex();
48  //!\endcode
49  //!
50 {

168 //! Atomically unlock associated mutex and wait until the condition
169 //! variable is signaled. Relock the associated mutex before return.
170 //! Wait forever if necessary. Return true if successful. The associated
171 //! mutex must be locked before invoking this method. Behavior is
172 //! unpredictable otherwise. Due to inconsitencies among various
173 //! implementations, do not assume this wait can be interrupted.
174 inline bool CondVar::wait()
175 {
176  return pthread_cond_wait(&cv_, &mu_) == 0;
