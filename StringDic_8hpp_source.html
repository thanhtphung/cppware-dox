19 //! dictionary of key-value strings
20 class StringDic
21  //!
22  //! A class representing a dictionary of key-value strings. Key-value pairs
23  //! are added using the add() and associate() methods, and are removed using
24  //! the rm() methods. Searches are provided by the find() methods. Implemented
25  //! using a 2-3-4 tree of StringPair instances. Example:
26  //!\code
27  //! StringDic dic;
28  //! :
29  //! dic.add(k0, v0); //add k0-v0 key-value pair
30  //! dic.add(k1, v1); //add k1-v1 key-value pair
31  //! dic.associate(k0, v2); //update k0's value
32  //! dic.rm(k1); //remove k1 association
33  //!\endcode
34  //!
