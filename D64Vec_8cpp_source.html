107 //!
108 //! Find and return the kth smallest item. Given items can be arbitrarily rearranged.
109 //! Behavior is unpredictable if k is not less than the number of items. The original
110 //! code is in public domain (algorithm by N. Wirth, implementation by N. Devillard).
111 //!
112 D64Vec::item_t D64Vec::findKthSmallest(item_t* item, size_t itemCount, size_t k)
113 {
114  for (size_t left = 0, right = itemCount - 1; left < right;)
115  {
116  item_t kthItem = item[k];
117  size_t i = left;
118  size_t j = right;
119  do
120  {
121  for (; item[i] < kthItem; ++i);
122  for (; item[j] > kthItem; --j);
123  if (i <= j)
124  {
125  item_t tmp = item[i];
126  item[i++] = item[j];
127  item[j--] = tmp;
128  }
129  } while (i <= j);
130  if (j < k) left = i;
131  if (i > k) right = j;
132  }
133 
134  return item[k];
135 }

331 //!
332 //! Assume vector is sorted, locate given item using binary search.
333 //! Return true if found (also return the located index in foundIndex).
334 //! Return false otherwise (also return the nearest item in foundIndex).
335 //! Behavior is unpredictable if vector is unsorted.
336 //!
337 bool D64Vec::search(item_t item, size_t& foundIndex) const
338 {
339 
340  // Assume not found.
341  bool found = false;
342 
343  // Locate item using binary search.
344  int loI = 0;
345  int hiI = static_cast<int>(numItems_ - 1);
346  unsigned int midI = 0;
347  while (loI <= hiI)
348  {
349  midI = (loI + hiI) >> 1;
350  if (item > item_[midI])
351  {
352  loI = midI + 1;
353  }
354  else if (item < item_[midI])
355  {
356  hiI = midI - 1;
357  }
358  else
359  {
360  found = true;
361  break;
362  }
363  }
364 
365  // Return true if found.
366  foundIndex = midI;
367  return found;
368 }

395  if (minCap > curCap)
