85 //!
86 //! Return ASCII form of this set using given range delimiter. ASCII form
87 //! of an empty set is an empty string. ASCII form of a non-empty set has
88 //! the following format: "$lo(-$hi)?[$rangeDelimiter$lo(-$hi)?]*". Sample
89 //! ASCII form with comma as range delimiter:
90 //!\code
91 //! 172.16.16.2-172.16.16.3,172.16.16.11,192.168.0.15-192.168.0.21
92 //!\endcode
93 //! Sample ASCII form with newline as range delimiter:
94 //!\code
95 //! 172.16.16.2-172.16.16.3
96 //! 172.16.16.11
97 //! 192.168.0.15-192.168.0.21
98 //!\endcode
99 //!
100 String IpAddrSet::toString(const char* delim) const
101 {
102  char buf[1 + IpAddr::StrLength + 1];
103  buf[0] = '-';
104 
105  IpAddr ipAddr;
106  String rangeDelim(delim);
107  String s;
108  for (size_t i = 0, numRanges = U32Set::numRanges(); i < numRanges; s += rangeDelim, ++i)
109  {
110  unsigned int loAddr;
111  unsigned int hiAddr;
112  U32Set::getRange(i, loAddr, hiAddr);
113  ipAddr = loAddr;
114  s += ipAddr.toString(buf + 1);
115  if (loAddr != hiAddr)
116  {
117  ipAddr = hiAddr;
118  ipAddr.toString(buf + 1);
119  s += buf;
120  }
121  }
122 
123  s.truncate(s.length() - rangeDelim.length());
124  return s;
125 }

129 // Form a U32Set string from the given IpAddrSet string. Save result in result. Return
130 // reference to result. Example: result for "172.16.16.1,172.16.16.90-171.16.16.99"
131 // is "2886733825,2886733914-2886733923".
132 //
133 String& IpAddrSet::normalize(String& result, const char* s, char delim) const
134 {
135 
136  // Look at input string as delimited text.
137  validateDelim(delim);
138  bool makeCopy = false;
139  DelimitedTxt txt(s, strlen(s), makeCopy, delim);
140 
141  // Normalize one line at a time.
142  IpAddr ipAddr;
143  String line;
144  bool trimLine = true;
145  char buf[U32::MaxDigits + 1 + U32::MaxDigits + 1];
146  result.reset();
147  while (txt.next(line, trimLine))
148  {
149  line.trimSpace(true /*trimL*/, true /*trimR*/);
150  size_t n;
151  unsigned int hiAddr;
152  unsigned int loAddr;
153  if (ipAddr.reset(line.ascii()))
154  {
155  n = U32::toDigits(ipAddr, buf);
156  }
157  else if (strToRange(line.ascii(), loAddr, hiAddr))
158  {
159  size_t n0 = U32::toDigits(loAddr, buf);
160  buf[n0] = '-';
161  size_t n1 = U32::toDigits(hiAddr, buf + n0 + 1);
162  n = n0 + 1 + n1;
163  }
164  else
165  {
166  continue;
167  }
168  buf[n++] = ',';
169  result.append(buf, n);
170  }
171 
172  // Return reference to <result>.
173  result.truncate(result.length() - 1);
174  return result;
175 }

266 //!
267 //! Convert string s to loAddr and hiAddr. Return true if s specifies a valid
268 //! dotted IP address range. A dotted IP address range is valid if the low address
269 //! is less than or equal to the high address.
270 //!
271 bool IpAddrSet::strToRange(const char* s, unsigned int& loAddr, unsigned int& hiAddr)
272 {
273 
274  // Assume string s has format "a.b.c.d-e.f.g.h" (e.g., "172.16.16.1-172.16.16.23").
275  char* p;
276  unsigned int abcd[8];
277  for (size_t i = 0; i < 8; ++i, s = p + 1)
278  {
279  abcd[i] = strtoul(s, &p, 0);
280  static const char s_delimiter[8] = {'.', '.', '.', '-', '.', '.', '.', 0};
281  if ((p == s) || (*p != s_delimiter[i]) || (abcd[i] > 255))
282  {
283 
284  // Special case for our CIDR notation. A format of "a.b.c.d/(0-32)"
285  // (e.g., "172.16.16.0/24") is required. This is slightly different
286  // compared to some tool like nmap. For example, an older nmap allowed
287  // "a.b.c[.d]/(0-32)", and the current nmap allows "a.b.c[.d]/(1-32)".
288  // TODO: be consistent with the new Subnet class.
289  if ((i == 3) && (p != s) && (*p == '/'))
290  {
291  s = p + 1;
292  unsigned int numBits = strtoul(s, &p, 0);
293  if ((p != s) && (*p == 0) && (numBits <= 32))
294  {
295  unsigned int mask = ((1U << numBits) - 1) << (32 - numBits);
296  loAddr = ((abcd[0] << 24) | (abcd[1] << 16) | (abcd[2] << 8) | abcd[3]) & mask;
297  hiAddr = loAddr + ~mask;
298  return true;
299  }
300  }
301 
302  // Not an IP address range.
303  return false;
304  }
305  }
306 
307  // Return true if <s> specifies a valid dotted IP address range.
308  loAddr = (abcd[0] << 24) | (abcd[1] << 16) | (abcd[2] << 8) | abcd[3];
309  hiAddr = (abcd[4] << 24) | (abcd[5] << 16) | (abcd[6] << 8) | abcd[7];
310  return (loAddr <= hiAddr);
311 }
