414 //
415 // Main loop for the messenger thread. Each thread has its own queue allowing
416 // commands/events to be delivered sequentially on a FIFO basis. The loop can
417 // terminate due to an interrupt or a null message.
418 //
419 void Messenger::loop()
420 {
421  ItemQ::Item* item;
422  unsigned int timeoutInMsecs = ItemQ::ETERNITY;
423  while ((!Thread::isTerminating()) && cmdQ_->get(item, timeoutInMsecs))
424  {
425 
426  // Allow messenger loop termination via a null message.
427  const Package* package = dynamic_cast<const Package*>(item);
428  if (package == 0)
429  {
430  break;
431  }
432 
433  // Send events sequentially.
434  const String& cmd = package->cmd();
435  if (cmd.empty())
436  {
437  const Paddr& thirdParty = dbuggee();
438  sndEvent(package->event(), thirdParty);
439  delete package;
440  continue;
441  }
442 
443  // Run commands sequentially.
444  // In synchronous mode, notify waiter by incrementing a semaphore.
445  // Package is to be deleted by waiter.
446  rsp_.reset();
447  runCmd(cmd);
448  Semaphore* sem = package->sem();
449  if (sem != 0)
450  {
451  *package->rsp() = rsp_;
452  sem->increment();
453  continue;
454  }
455 
456  // In asynchronous mode, invoke onRsp callback upon each command completion.
457  void* arg;
458  onRsp_t onRsp = package->onRsp(arg);
459  onRsp(arg, cmd, rsp_);
460  delete package;
461  }
462 }
