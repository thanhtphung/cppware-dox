150 //
151 // exit [--crash ]
152 // [--restart] #allowed if run as a service
153 //
154 bool NetDemonCmd::doExit(const CmdLine& req)
155 {
156 
157  // Determine exiting scheme.
158  String optK("crash");
159  Bool crash(req.opt(optK), false /*defaultV*/);
160  optK = "restart";
161  const String* restartOpt = req.opt(optK);
162  Bool restart(restartOpt, false /*defaultV*/);
163 
164  // Sanity check.
165  bool ok = true;
166  const char* rsp;
167  if (restart)
168  {
169  if (netDemon_->runningAsProcess())
170  {
171  ok = false;
172  rsp = "exit: restarting a process is unsupported.";
173  }
174  else
175  {
176  rsp = crash? "Crashing and restarting...": "Restarting...";
177  }
178  }
179  else
180  {
181  rsp = crash? "Crashing...": "Exiting...";
182  }
183 
184  // Perform operation if allowed.
185  respond(req, rsp);
186  Thread::yield();
187  if (ok)
188  {
189  if (crash)
190  {
191  if (restartOpt != 0)
192  {
193  netDemon_->setRestartMode(restart);
194  }
195  const char* const TOMBSTONE = "NetDemonCmd::doExit: crashing...";
196  const char** grave = 0;
197  *grave = TOMBSTONE;
198  }
199  netDemon_->stop(restart);
200  }
201 
202  bool cmdIsValid = true;
203  return cmdIsValid;
204 }
