115 bool UnitTestBed::run(const char* registryName) const
116 {
117  std::string key((registryName == 0)? DEFAULT_REGISTRY: registryName);
118  CppUnit::TestFactoryRegistry& registry = CppUnit::TestFactoryRegistry::getRegistry(key);
119  CppUnit::Test* testSuite = registry.makeTest();
120  CppUnit::TestRunner runner;
121  runner.addTest(testSuite);
122 
123  UnitTestStat result(beQuiet_, beVerbose_, display_);
124  bool ok = true;
125  size_t numArgs = cmdLine_.numArgs();
126  if (numArgs > 1)
127  {
128  std::string testName;
129  for (unsigned int i = 1; i < numArgs; ++i)
130  {
131  const String* arg = cmdLine_.arg(i);
132  testName.assign(arg->ascii(), arg->byteSize() - 1);
133  display_ << "running=";
134  display_ << *arg;
135  display_ << "...";
136  display_ << std::endl;
137  try
138  {
139  runner.run(result, testName);
140  }
141  catch (std::invalid_argument& /*e*/)
142  {
143  display_ << *arg;
144  display_ << ": skipped (unknown test name)";
145  display_ << std::endl;
146  ok = false;
147  }
148  }
149  }
150 
151  else
152  {
153  display_ << "running=all...";
154  display_ << std::endl;
155  runner.run(result);
156  }
157 
158  display_ << "run=";
159  display_ << result.numStarted();
160  display_ << " ok=";
161  display_ << result.numOkayed();
162  display_ << " failed=";
163  display_ << result.numFailed();
164  display_ << std::endl;
165  return (ok && (result.numFailed() == 0));
