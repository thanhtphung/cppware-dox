18  //!
19  //! A class representing a UTF8 character. Natively, a UTF8 character
20  //! has a 32-bit value with range 0x000000..0x00d7ff,0x00e000..0x10ffff.
21  //! When encoded, depending on its value, up to 4 bytes are required.
22  //! Construct an instance to encode/decode a UTF8 character. This
23  //! implementation matches the newer UTF8 definition as described
24  //! in RFC 3629. The older UTF8 definition is less restrictive and
25  //! is described in RFC 2279. Example:
26  //!\code
27  //! Utf8 c0(0x54321U);
28  //! utf8_t seq[Utf8::MaxSeqLength];
29  //! unsigned int seqLength = c0.encode(seq);
30  //! Utf8 c1;
31  //! if ((c1.decode(seq, seqLength) != seqLength) || (c1.asU32() != c0.asU32()))
32  //! {
33  //! assert("coding error" == 0);
34  //! }
35  //!\endcode
36  //!
37 {

230 //! Return the actual UTF8 byte sequence length for this instance if encoded.
231 //! A UTF8 character requires between MinSeqLength and MaxSeqLength bytes.
232 //! Basically, more bytes are required for larger values.
233 inline unsigned int Utf8::getSeqLength() const
234 {
235  const unsigned int MAX_VALUE_2 = 0x000007ffU; //11-bit, 2 UTF8 bytes
236  const unsigned int MAX_VALUE_3 = 0x0000ffffU; //16-bit, 3 UTF8 bytes
237  if (c_ <= MaxAscii) return 1;
238  if (c_ <= MAX_VALUE_2) return 2;
239  if (c_ <= MAX_VALUE_3) return 3;
240  return 4;
241 }
