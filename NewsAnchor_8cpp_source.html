103 // Remove given source from the collection.
104 bool NewsSource::rm(NewsSubject* source)
105 {
106  unsigned int index = source->indexInSource();
107  bool maintainOrder = false;
108  bool ok = Vec::rmFromIndex(index, maintainOrder);
109  if (ok)
110  {
111  if (strlen(source->name()) == maxNameLength_)
112  {
113  updateMaxNameLength();
114  }
115  source->setIndexInSource(INVALID_INDEX);
116  if (index < Vec::numItems())

119  source->setIndexInSource(index);

121  }

284 String NewsAnchor::formHeadline(const NewsSubject& source, const NewsStory& story)
285 {
286 
287  // Form timestamp (example: "|10/25/11 15:46:11|123456| ").
288  struct tm now;
289  time_t t = time(0);
290  localtime_s(&now, &t);
291  char stamp[32 + U64::MaxDigits + 2 + 1];
292  size_t size = strftime(stamp, 32, "|%c|", &now);
293  size += U64::toDigits(story.timeAsFiletime(), stamp + size);
294  stamp[size++] = '|';
295  stamp[size++] = ' ';
296  stamp[size++] = 0;
297 
298  String logEntry(stamp, size);
299  logEntry += source.name();
300  logEntry += " #";
301  logEntry += story.summary();
302 
303  // A fabricated news story has a double-null terminated summary.
304  // Make it look normal as a log entry.
305  if (*(logEntry.raw() + logEntry.byteSize() - 2) == 0)
306  {
307  logEntry.truncate(logEntry.length() - 1);
308  }
309 
310  logEntry += NEW_LINE;
311  return logEntry;
312 }

339 //!
340 //! Iterate the registry. Invoke callback at each known news source. The callback
341 //! should return true to continue iterating and should return false to abort
342 //! iterating. Return false if the callback aborted the iterating. Return true
343 //! otherwise. Behavior is unpredictable if callback attempts updating the registry
344 //! during the iterating.
345 //!
346 bool NewsAnchor::apply(cb0_t cb, void* arg) const
347 {
348  bool ok = true;

350  SpinSection::Lock lock(ss_);
351  for (size_t i = 0, numSources = newsSource_->numItems(); i < numSources; ++i)
352  {
353  NewsSubject* source = newsSource_->peek(i);
354  if (!cb(arg, source))
355  {
356  ok = false;
357  break;
358  }
359  }
360  }
361 
362  // Return true if the iterating was not aborted.
363  return ok;

493 void NewsAnchor::createDefaultHeadlines()
494 {
495 
496  // Allow disabling default news (Cmd/NewsSubject availability) via an environment variable.
497  // If __AppkitEnableDefaultNews does exist, use it to determine if default news should be
498  // disabled. By default, Cmd and NewsSubject availablility is announced by the news anchor.
499  // Note that cmdNews_ and sourceNews_ are built-in news subjects owned by the news anchor.
500  // They need to be constructed/destructed without impacting the anchor reference count so
501  // that the news anchor can be destructed via reference counting.
502  String k(ENABLE_DEFAULT_NEWS);
503  String v;
504  bool enableDefaultNews = (!getenv(k, v)) || Bool(v);
505  if (enableDefaultNews)
506  {
507 
508  // Monitor for command map updates so command availability/unavailability can be
509  // announced. Some commands have already been created before this monitoring, so
510  // try reporting them also. The reporting of those commands is a little questionable
511  // as some/all of them can still exist after the news anchor destruction.
512  int qGrowth = -1;
513  cmdNews_ = NewsSubject::createHeadline(CMD_NEWS_NAME, CMD_NEWS_Q_CAP, qGrowth);
514  this->rmRef(); //special case for cmdNews_ construction
515  CmdMap* cmdMap = cmd_->map();
516  cmdMap->apply(announceCmdAvailability, this);
517  cmdMap->monitor(onCmdMapUpdate, this, &oldCmdMapUpdateCb_, &oldCmdMapUpdateArg_);
518 
519  // Source news availablity cannot be announced in addSource() as the observable does
520  // not exist there yet. Announce its availability here after it's been constructed.
521  sourceNews_ = NewsSubject::createHeadline(SOURCE_NEWS_NAME, SOURCE_NEWS_Q_CAP, qGrowth);
522  this->rmRef(); //special case for sourceNews_ construction
523  String summary("source --add ");
524  summary += sourceNews_->name();
525  sourceNews_->update(summary);
526  }
527 }

554 //
555 void NewsAnchor::loop()
556 {
557  createDefaultHeadlines();
558  defaultHeadlinesAreReady_ = true;
559 
560  const char* name = "-news-";
561  LogPath logPath(name, LOG_PATH_LO, LOG_PATH_HI);
562  curLog_ = logPath.curLog();
563 
564  ItemQ::Item* item;
565  unsigned int storyCount = 0;
566  unsigned int timeoutInMsecs = ItemQ::ETERNITY;
567  while ((!Thread::isTerminating()) && toBeAiredQ_->get(item, timeoutInMsecs))
568  {
569  NewsSubject* news = dynamic_cast<NewsSubject*>(item);
570

572  {
573  const NewsStory* story = news->getStory();
574  news->notify(story->timeAsFiletime(), story->summary());
575  logHeadline(*news, *story, curLog_);
576  news->nullifyStory();
577  }
578  catch (...)
579  {
580  }
581 
582  news->rmRef();
583 
584  if ((++storyCount % LOG_FILE_CAP) == 0)

586  curLog_ = logPath.reopen();
587  }
588  }
