50 //!
51 //! Locate given singleton. Return the per-process singleton. If non-existent, return zero.
52 //!
53 Singleton* Singleton::getSingleton(const char* id)
54 {
55  CriSection* cs;
56  const Foundation& r = Foundation::instance();
57  Vec* vec = r.singletonVec(cs);
58  CriSection::Lock lock(*cs);
59 
60  size_t foundIndex;
61  bool found = vec->find(id, diff, foundIndex);
62  Singleton* singleton = found? static_cast<Singleton*>(vec->peek(foundIndex)): 0;
63  return singleton;
64 }
