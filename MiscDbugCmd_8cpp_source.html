234 bool MiscDbugCmd::doEnv(const CmdLine& req)
235 {
236  wchar_t* envW = GetEnvironmentStringsW();
237  StringVec env;
238 
239  String optK("expandEnv");
240  bool expandEnv = Bool(req.opt(optK), false /*defaultV*/);
241  DelimitedTxt txt(EQUALS_SIGN);
242  String k;
243  String kv;
244  String v;
245  for (const wchar_t* w = envW; *w;)
246  {
247  kv = w;
248  txt.setTxt(kv, false /*makeCopy*/);
249  (txt.next(k, true /*doTrimLine*/) && getenv(k, v, expandEnv))?
250  env.add(k + EQUALS_SIGN + v):
251  env.add(kv);
252  size_t n = wcslen(w);
253  w += n + 1;
254  }
255 
256  FreeEnvironmentStringsW(envW);
257  env.sort();
258  String rsp(env.stringify(NEW_LINE));
259  respond(req, rsp);
260 
261  bool cmdIsValid = true;
262  return cmdIsValid;
263 }

297 bool MiscDbugCmd::getAddrRange(const CmdLine& req, IpAddrSet& addrRange)
298 {
299  addrRange.reset();
300  size_t numArgs = req.numArgs();
301  for (size_t i = 1; i < numArgs; ++i)
302  {
303  const String* arg = req.arg(i);
304  char delim = ',';
305  if (!IpAddrSet::isValid(arg->ascii(), delim))
306  {
307  addrRange.reset();
308  break;
309  }
310  addrRange.add(arg->ascii(), delim);
311  }
312 
313  bool ok = (addrRange.numAddrs() > 0);
314  return ok;
315 }
