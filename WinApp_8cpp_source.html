81 //
82 // This is the backup message handler for WinApp. Messages are usually intercepted
83 // and handled in the runMessageLoop() messaging loop. However, that loop can be
84 // bypassed when the UI thread is modal (e.g., user is moving/resizing a window).
85 // In that scenario, messages can still be dispatched via a message loop internal
86 // to the modal system. This is why PostMessage() and not PostThreadMessage() must
87 // be used to communicate with the WinApp messaging thread.
88 //
89 LRESULT CALLBACK WinApp::WndProc(HWND hwnd, unsigned int msg, WPARAM wparam, LPARAM lparam)
90 {
91  WinApp* winApp = instance();
92  if (msg == winApp->messagingId_)
93  {
94  WinCmd* cmd = reinterpret_cast<WinCmd*>(wparam);
95  const CmdLine* req = reinterpret_cast<const CmdLine*>(lparam);
96  size_t i = Cmd::cmdIndex(*req);
97  cmd->cmdIsValid_ = cmd->onFinalRun(*req);
98  cmd->sem_[i].increment();
99  LRESULT lr = 0;
100  return lr;
101  }
102 
103  if (msg == WM_QUIT)
104  {
105  winApp->exitCode_ = wparam;
106  winApp->exitMessageLoop_ = true;
107  }
108 
109  LRESULT lr = DefWindowProc(hwnd, msg, wparam, lparam);
110  return lr;
111 }

212 //!
213 //! This method is invoked when the application is to start running. The method
214 //! is expected to return an exit code. The default implementation initiates
215 //! start-up command processing. A basic window messaging loop can also be entered
216 //! here if the application is configured to do so via setDefaultMessageLoop().
217 //! Start-up commands reside in corresponding rc3 file located in the confDir()
218 //! directory with name "Path(path()).base().rc3". For example, xxx.rc3 would be
219 //! the corresponding rc3 file for xxx.exe. The rc3 file is optional.
220 //!
221 int WinApp::onRun()
222 {
223  bool keepGoing = onWalk();
224  if (!keepGoing)
225  {
226  int rc = InitFailed;
227  return rc;
228  }
229 
230  // Allow users to implement their own onWalk() version. Initiate start-up
231  // command processing. Run the commands using a separate thread as they
232  // might need to interact with this thread which will enter a basic window
233  // messaging loop momentarily.
234  if (sysIo_ != 0)
235  {
236  const MappedTxtFile* rcFile = mapRcFile();
237  if (rcFile->isOk())
238  {
239  runArg_t* arg = new runArg_t(this, rcFile);
240  unsigned int stackSizeInBytes = 0; //default
241  bool startSuspended = true;
242  Thread* t = new Thread(agentEntry, arg, stackSizeInBytes, startSuspended);
243  arg->t = t;
244  if ((!t->isOk()) || (!t->resume()))
245  {
246  delete arg;
247  delete t;
248  }
249  else
250  {
251  rcFile = 0; //to be deleted by agent thread at end of start-up command processing
252  }
253  }
254  delete rcFile;
255  }
256 
257  int rc = enterLoopAtRun_? runMessageLoop(): Ok;
258  return rc;
259 }

273 //!
274 //! Custom windows messaging loop. Allow WinCmd commands to be handled by the
275 //! messaging thread. Basically, WinCmd::onRun() is invoked by a dedicated agent
276 //! task. That task then waits for WinCmd::onFinalRun() to be invoked by this
277 //! windows messaging thread. A semaphore is used to synchronize the two agents.
278 //! A derived class can override this to implement its own loop. To support WinCmd
279 //! command handling, application code must still dispatch windows messages timely
280 //! using its own scheme.
281 //!
282 int WinApp::runMessageLoop()
283 {
284 
285  // The loop terminates when a WM_QUIT message is received or if GetMessageW() fails.
286  int rc;
287  size_t oldMessagingThreadId = messagingThreadId_;
288  messagingThreadId_ = Thread::myId();
289  MSG msg = {0};
290  while (((rc = GetMessageW(&msg, 0, 0, 0)) > 0) && (!exitMessageLoop_))
291  {
292 
293  if (msg.message == messagingId_)
294  {
295  WinCmd* cmd = reinterpret_cast<WinCmd*>(msg.wParam);
296  const CmdLine* req = reinterpret_cast<const CmdLine*>(msg.lParam);
297  size_t i = Cmd::cmdIndex(*req);
298  cmd->cmdIsValid_ = cmd->onFinalRun(*req);
299  cmd->sem_[i].increment();
300  continue;
301  }
302 
303  TranslateMessage(&msg);
304  DispatchMessage(&msg);
305  }
306 
307  // WM_QUIT received in the backup message handler: WndProc().
308  if (exitMessageLoop_)
309  {
310  msg.wParam = exitCode_;
311  rc = 0;
312  }
313 
314  messagingThreadId_ = oldMessagingThreadId;
315  rc = (rc == 0)? static_cast<int>(msg.wParam): Fatal;
316  return rc;
317 }

343 // Run start-up commands from rc3 file, if any.
344 // This is done in the background by a separate thread.
345 //
346 void WinApp::runStartUpCmds(const MappedTxtFile& rcFile) const
347 {
348 
349  // Run start-up commands from rc3 file using loopback agent.
350  bool makeCopy = false;
351  DelimitedTxt rcTxt(reinterpret_cast<const char*>(rcFile.image()), static_cast<size_t>(rcFile.imageSize()), makeCopy);
352  String cmd;
353  bool doTrimLine = true;
354  while (rcTxt.next(cmd, doTrimLine) && (!cmd_->exitInProgress()))
355  {
356  if (!cmd.empty())
357  {
358  loopbackAgent_->runCmd(cmd);
359  }
360  }
