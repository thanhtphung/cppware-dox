241 const String& Path::shrink(String& fullName)
242 {
243 
244  // Split full name into subdirectories.
245  // Optimize out the "." and ".." parts.
246  String line;
247  StringVec parts;
248  DelimitedTxt txt(fullName, false, Directory::MARKER);
249  while (txt.next(line))
250  {
251  if ((line.byteSize() == 2) || ((line.byteSize() == 3) && (*line.ascii() == DOT))) //"/" or "./"
252  {
253  continue;
254  }
255  if (line == "../")
256  {
257  if (parts.rmTail())
258  {
259  continue;
260  }
261  }
262  parts.add(line);
263  }
264 
265  // Check for UNC path (//server/xxx).
266  const char* s = fullName.ascii();
267  bool isUnc = (s[0] == Directory::MARKER) && (s[1] == Directory::MARKER) && (s[2] != Directory::MARKER);
268  isUnc? fullName.reset(2, Directory::MARKER): fullName.reset();
269 
270  // Merge subdirectories back into a full name.
271  for (size_t i = 0, numParts = parts.numItems(); i < numParts; ++i)
272  {
273  fullName += parts.peek(i);
274  }
275 
276  return fullName;
277 }
