59 //!
60 //! Activate observer. That is, attach and follow the news subject of interest. The subject was
61 //! specified at construction. Invoke onAttach() and onDetach() when the subject attach/detach
62 //! occurs. With this invocation, the observer is now considered active. This method is a no-op
63 //! for active observers. That is, a subsequent activate() is meaningless without some deactivate()
64 //! in between.
65 //!
66 void NewsObserver::activate()
67 {
68  unsigned int oldV;
69  active_.set(1U, oldV);
70  if (oldV) //already active
71  {
72  return;
73  }
74 
75  // Try attaching now in case the subject is already available.
76  sourceWatch_->activate();
77  NewsSubject* attached = NewsSubject::attachByName(this, subject_->ascii());
78  if (attached)
79  {
80  attached_.set(1U, oldV);
81  if (oldV == 0U)
82  {
83  onAttach(attached);
84  }
85  }
86 }

89 //!

91 //! specified at construction. Also invoke onDetach() if the subject detaching occurs now.
92 //! With this invocation, the observer is now considered idle. This method is a no-op for
93 //! idle observers. That is, a subsequent deactivate() is meaningless without some activate()
94 //! in between.
