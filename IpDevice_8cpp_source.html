281 unsigned int /*IpDevice::addrCategory_e*/ IpDevice::categorize(unsigned int addr)
282 {
283 
284  // Loopback?
285  // Broadcast?
286  initialize();
287  if ((addr >> 24) == LoopbackNet)
288  {
289  return ((addr == 0x7f000000U) || (addr == 0x7fffffffU))? Broadcast: Loopback;
290  }
291 
292  const IpDevice* const* p = device_;
293  const IpDevice* const* pEnd = p + numDevices_;
294  unsigned int category = Far;
295  for (; p < pEnd; ++p)
296  {
297  const IpDevice* device = *p;
298  unsigned int numAddrs = device->numAddrs_;
299  for (unsigned int i = 0; i < numAddrs; ++i)
300  {
301 
302  // Local? (home address of one of the IP devices)
303  if (addr == device->addr_[i])
304  {
305  return Local;
306  }
307 
308  // Broadcast? (broadcast address for a connected subnet)
309  unsigned int netMask = device->netMask_[i];
310  unsigned int broadcast0 = device->addr_[i] & netMask;
311  unsigned int broadcast1 = broadcast0 | ~netMask;
312  if ((addr == broadcast0) || (addr == broadcast1))
313  {
314  return Broadcast;
315  }
316 
317  // Near? (part of a connected subnet)
318  // Only if it's not a local address nor a broadcast address.
319  if ((category != Near) && (addr & netMask) == broadcast0)
320  {
321  category = Near;
322  }
323  }
324  }
325 
326  return category;
327 }
