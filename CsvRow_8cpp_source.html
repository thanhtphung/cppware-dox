63 //!
64 //! Iterate the row from left to right. Invoke callback at each column. The callback
65 //! should return true to continue iterating and should return false to abort iterating.
66 //! Return false if the callback aborted the iterating. Return true otherwise.
67 //!
68 bool CsvRow::apply(cb1_t cb, void* arg) const
69 {
70 
71  // Iterate from left to right. Invoke callback at each column.
72  // Return immediately if aborting.
73  const char delim = delim_;
74  const utf8_t* p;
75  const utf8_t* p1;
76  char quote = 0;
77  for (p = p1 = row_; p < pEnd_; ++p)
78  {
79  if (quote == 0)
80  {
81  if (p[0] == delim)
82  {
83  if (!cb(arg, reinterpret_cast<const char*>(p1), p - p1 + 1))
84  {
85  return false;
86  }
87  p1 = p + 1;
88  }
89  else if ((p[0] == DOUBLE_QUOTE) || (p[0] == SINGLE_QUOTE))
90  {
91  quote = p[0];
92  }
93  }
94  else if ((p[0] == quote) && (p[-1] != BACKSLASH))
95  {
96  quote = 0;
97  }
98  }
99 
100  // Still need to take care of the last column w/o a column delimiter.
101  if (p1 < p)
102  {
103  return cb(arg, reinterpret_cast<const char*>(p1), p - p1);
104  }
105 
106  // Return true to indicate the iterating was not aborted.
107  return true;
108 }

111 //!
112 //! Remove surrounding quotes from given column. Also resolve escape sequences
113 //! treating the column as a C++ string literal. As an exception, replace two
114 //! consecutive quotes, if any, with one. No-op if given column is not quoted.
115 //! Return true if column was modified (due to surrounding quote removal and/or
116 //! escape sequence resolution). For a single quoted column, escape sequence
117 //! resolution can be skipped if unescapeSingleQuotedString is false.
118 //!
119 bool CsvRow::dequoteCol(String& col, bool unescapeSingleQuotedString)
120 {
121 
122  // Quoted string.
123  bool modified;
124  const char* p = col.ascii();
125  const char* pLast = p + col.byteSize() - 2;
126  if ((col.length() >= 2) && ((*p == DOUBLE_QUOTE) || (*p == SINGLE_QUOTE)) && (p < pLast) && (*pLast == *p))
127  {
128  --pLast;
129  for (char quote = *p++; p < pLast; ++p)
130  {
131  if ((p[0] == quote) && (p[1] == quote))
132  {
133  size_t index = p - col.ascii();
134  col.setAscii(index, BACKSLASH); //"\"\"" --> "\\\"" and "\'\'" --> "\\\'"
135  p = col.ascii() + index + 1;
136  pLast = col.ascii() + col.byteSize() - 2 - 1;
137  }
138  }
139  modified = col.dequote(unescapeSingleQuotedString);
140  }
141 
142  // No surrounding quotes.
143  else
144  {
145  modified = false;
146  }
147 
148  // Return true if column was modified.
149  return modified;
150 }

202 //!
203 //! Retrieve the next column. Return true if there's one. Otherwise, return
204 //! false and zeroes in col and length (if row is empty or if there's no more
205 //! columns). The first invocation after construction or reset() will return
206 //! the first column.
207 //!
208 bool CsvRow::next(const char*& col, size_t& length)
209 {
210 
211  // Assume unsuccessful.
212  bool ok = false;
213 
214  // See if first column is available.
215  if (p1_ == 0)
216  {
217  if (row_ < pEnd_)
218  {
219  ok = true;
220  p1_ = row_;
221  }
222  }
223 
224  // See if next column is available.
225  else if ((p2_ + 1) < pEnd_)
226  {
227  ok = true;
228  p1_ = p2_ + 1;
229  }
230 
231  // Retrieve the next column.
232  // If a column delimiter is in a quoted string, consider it part of the string.
233  if (ok)
234  {
235  const utf8_t* p;
236  const utf8_t* pLast;
237  const char delim = delim_;
238  char quote = 0;
239  for (p = p1_, pLast = pEnd_ - 1; p < pLast; ++p)
240  {
241  if (quote == 0)
242  {
243  if (p[0] == delim) break;
244  else if ((p[0] == DOUBLE_QUOTE) || (p[0] == SINGLE_QUOTE)) quote = p[0];
245  }
246  else if ((p[0] == quote) && (p[-1] != BACKSLASH))
247  {
248  quote = 0;
249  }
250  }
251  p2_ = p;
252  col = reinterpret_cast<const char*>(p1_);
253  length = p - p1_ + 1;
254  }
255  else
256  {
257  col = 0;
258  length = 0;
259  }
260 
261  // Return true if successful.
262  return ok;
263 }

282 //!
283 //! Return the total number of columns in the row. Return zero if the attached
284 //! row is empty. A column consists of consecutive characters terminated by the
285 //! column delimiter. The last column can but does not have to be delimited by
286 //! the column delimiter. That is, non-empty row consists of at least one column
287 //! even if the row has no column delimiters.
288 //!
289 size_t CsvRow::countCols() const
290 {
291 
292  // Iterate from left to right. Count each column.
293  const char delim = delim_;
294  size_t numCols = 0;
295  char quote = 0;
296  for (const utf8_t* p = row_; p < pEnd_; ++p)
297  {
298  if (quote == 0)
299  {
300  if (p[0] == delim) ++numCols;
301  else if ((p[0] == DOUBLE_QUOTE) || (p[0] == SINGLE_QUOTE)) quote = p[0];
302  }
303  else if ((p[0] == quote) && (p[-1] != BACKSLASH))
304  {
305  quote = 0;
306  }
307  }
308 
309  // Still need to take care of the last column w/o a column delimiter.
310  if ((row_ != pEnd_) && (*(pEnd_ - 1) != delim))
311  {
312  ++numCols;
313  }
314 
315  // Return the total number of columns in the text.
316  return numCols;
317 }
