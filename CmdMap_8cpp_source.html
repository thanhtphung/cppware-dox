102 //!
103 //! Iterate the map. Invoke callback at each command. The callback should
104 //! return true to continue iterating and should return false to abort
105 //! iterating. Return false if the callback aborted the iterating. Return
106 //! true otherwise. The map must not be updated during the iterating. Behavior
107 //! is unpredictable if it is udpated directly or indirectly by the callback.
108 //!
109 bool CmdMap::apply(cb0_t cb, void* arg) const
110 {
111  bool ok = true;
112  {
113  CriSection::Lock lock(cs_);
114  for (size_t i = 0; i < map_.numItems(); ++i)
115  {
116  const kvPair_t* kv = static_cast<const kvPair_t*>(map_.peek(i));
117  if (!cb(arg, kv->name, kv->cmd, kv->cmdIndex))
118  {
119  ok = false;
120  break;
121  }
122  }
123  }
124 
125  // Return true if the iterating was not aborted.
126  return ok;
127 }

149 //!
150 //! Add given command handler to map. Each handler is responsible for at least
151 //! zero but not more than 256 commands. Return the number of commands successfully
152 //! mapped. Also return the number of erroneous names (duplicate or overlimit).
153 //!
154 unsigned int CmdMap::add(Cmd* cmd, unsigned int& numErrs)
155 {
156  bool ok = true;
157  unsigned int numCmds;
158  String cmdName;
159  StringVec added;
160  unsigned char cmdIndex = 0;
161  Tokenizer tokenizer(cmd->nameSet());
162  {
163  CriSection::Lock lock(cs_);
164  unsigned int numItems = map_.numItems();
165  while (tokenizer.next(cmdName))
166  {
167  size_t kvSize = offsetof(kvPair_t, name) + cmdName.length() + 1;
168  unsigned char* p = new unsigned char[kvSize];
169  kvPair_t* kv = reinterpret_cast<kvPair_t*>(p);
170  kv->cmd = cmd;
171  kv->cmdIndex = cmdIndex++;
172  memcpy(kv->name, cmdName.raw(), cmdName.byteSize());
173  if (!map_.addIfNotFound(kv))
174  {
175  ok = false;
176  delete[] p;
177  }
178  else
179  {
180  added.add(cmdName);
181  if (cmdName.length() > maxNameLength_)
182  {
183  maxNameLength_ = cmdName.length();
184  }
185  }
186  if (cmdIndex == 0)
187  {
188  if (tokenizer.peek())
189  {
190  ok = false;
191  }
192  break;
193  }
194  }
195  numCmds = map_.numItems() - numItems;
196  }
197 
198  // Announce command availability.
199  bool available = true;
200  for (size_t i = 0, addCount = added.numItems(); i < addCount; ++i)
201  {
202  updateCb_(updateCbArg_, added.peek(i), available);
203  }
204 
205  // Return number of mapped and unmapped commands.
206  numErrs = ok? (0): (tokenizer.countTokens() - numCmds);
207  return numCmds;
208 }

211 //!
212 //! Remove given command handler from map. Each handler is responsible
213 //! for at least one but not more than 256 commands. Return the number
214 //! of commands successfully unmapped.
215 //!
216 unsigned int CmdMap::rm(const Cmd* cmd)
217 {
218  bool findMaxNameLength = false;
219  unsigned int numCmds;
220  String cmdName;
221  StringVec removed;
222  Tokenizer tokenizer(cmd->nameSet());
223  {
224  CriSection::Lock lock(cs_);
225  unsigned int numItems = map_.numItems();
226  while (tokenizer.next(cmdName))
227  {
228  void* removedItem;
229  if (map_.rm(cmdName.ascii(), compareName, removedItem))
230  {
231  removed.add(cmdName);
232  if (cmdName.length() == maxNameLength_)
233  {
234  findMaxNameLength = true;
235  }
236  delete[] static_cast<const unsigned char*>(removedItem);
237  }
238  }
239  numCmds = numItems - map_.numItems();
240  }
241 
242  // A command with a long name was removed,
243  // so find the length of the longest command name.
244  if (findMaxNameLength)
245  {
246  size_t maxNameLength = 0;
247  CriSection::Lock lock(cs_);
248  for (size_t i = 0; i < map_.numItems(); ++i)
249  {
250  const kvPair_t* kv = static_cast<const kvPair_t*>(map_.peek(i));
251  size_t nameLength = strlen(kv->name);
252  if (nameLength == maxNameLength_)
253  {
254  maxNameLength = maxNameLength_;
255  break; //no change since another command name is as long
256  }
257  if (nameLength > maxNameLength)
258  {
259  maxNameLength = nameLength;
260  }
261  }
262  maxNameLength_ = static_cast<unsigned int>(maxNameLength);
263  }
264 
265  // Announce command unavailability.
266  bool available = false;
267  for (size_t i = 0, removeCount = removed.numItems(); i < removeCount; ++i)
268  {
269  updateCb_(updateCbArg_, removed.peek(i), available);
270  }
271 
272  // Return number of commands successfully unmapped.
273  return numCmds;
274 }
