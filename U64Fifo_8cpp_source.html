54 //!
55 //! Copy the queue contents from given queue. This queue might grow to
56 //! accomodate the source. If it cannot grow, items at the head will be
57 //! dropped.
58 //!
59 const U64Fifo& U64Fifo::operator =(const U64Fifo& q)
60 {
61 
62  // Prevent self assignment.
63  if (this == &q)
64  {
65  return *this;
66  }
67 
68  // Might need to grow to accomodate source.
69  unsigned int minCap = q.numItems_;
70  if (minCap > capacity())
71  {
72  if (canGrow())
73  {
74  delete[] item_;
75  item_ = new item_t[setNextCap(minCap)];
76  }
77  }
78 
79  // Copy all items.
80  unsigned int curCap = capacity();
81  numRms_ += numItems_;
82  if (minCap <= curCap)
83  {
84  construct(q);
85  numAdds_ += numItems_;
86  if (numItems_ > usagePeak_)
87  {
88  usagePeak_ = numItems_;
89  }
90  }
91 
92  // Source queue has too many items. Drop items at the head.
93  else if (curCap > 0)
94  {
95  reset();
96  for (size_t i = q.numItems_ - curCap, iEnd = q.numItems_; i < iEnd; ++i)
97  {
98  item_t item = q.peek(i);
99  U64Fifo::add(item);
100  }
101  }
102 
103  // Return reference to self.
104  return *this;
105 }

246 //!
247 //! Resize queue. Given new capacity must not be less than the current
248 //! queue size. Return true if successful.
249 //!
250 bool U64Fifo::resize(unsigned int newCap)
251 {
252  bool ok;
253  unsigned int curCap = capacity();
254  if (numItems_ > newCap)
255  {
256  ok = false;
257  }
258 
259  // No-op if capacity does not change.
260  else if (newCap == curCap)
261  {
262  ok = true;
263  }
264 
265  // Resize.
266  else
267  {
268 
269  // Items are contiguous.
270  item_t* item = new item_t[newCap];
271  if (head_ < tail_)
272  {
273  memcpy(item, item_ + head_, numItems_ * sizeof(*item_));
274  }
275 
276  // Items wrap.
277  else if (numItems_ > 0)
278  {
279  size_t part0Count = curCap - head_;
280  memcpy(item, item_ + head_, part0Count * sizeof(*item_));
281  size_t part1Count = tail_;
282  memcpy(item + part0Count, item_, part1Count * sizeof(*item_));
283  }
284 
285  head_ = 0;
286  tail_ = numItems_;
287  delete[] item_;
288  item_ = item;
289  setCapacity(newCap);
290  ok = true;
291  }
292 
293  // Return true if successful.
294  return ok;
295 }
