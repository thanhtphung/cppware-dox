42  {
43  name = (ver.wProductType == VER_NT_WORKSTATION)? L"win-7": L"win-2008-r2";
44  }
45  else if (ver.dwMinorVersion == 0)

47  name = (ver.wProductType == VER_NT_WORKSTATION)? L"win-vista": L"win-2008";

201 //!
202 //! Apply given callback to the first maxPids running processes found.
203 //! The callback should return true to continue iterating and should
204 //! return false to abort iterating. Return false if the callback aborted
205 //! the iterating. Return true otherwise.
206 //!
207 bool Process::apply(unsigned int maxPids, cb1_t cb, void* arg)
208 {
209  bool ok = true;
210  DWORD* pids = new DWORD[maxPids];
211 
212  DWORD bufSizeIn = maxPids * sizeof(*pids);
213  DWORD bufSizeOut = 0;
214  BOOL success = EnumProcesses(pids, bufSizeIn, &bufSizeOut);
215  if (success)
216  {
217  Process proc(0UL);
218  for (unsigned int i = 0, numPids = bufSizeOut / sizeof(*pids); i < numPids; ++i)
219  {
220  proc.id_ = pids[i];
221  bool keepGoing = cb(arg, proc);
222  if (!keepGoing)
223  {
224  ok = false;
225  break;
226  }
227  }
228  }
229 
230  delete[] pids;
231  return ok;
232 }
