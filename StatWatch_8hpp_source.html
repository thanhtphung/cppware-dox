13 //! stop watch with stats tracking
14 class StatWatch
15  //!
16  //! A class representing a stop watch with stats tracking. A delta is
17  //! formed for every start and stop time pair. Deltas are accumulated
18  //! to provide the basic stats (min, max, and average) when requested.
19  //! The time stamps are 64-bit tick counts. The tick resolution (i.e.,
20  //! ticks per second) is platform dependent. Use TickTime to interpret
21  //! the tick counts. If there's no need for the provided stats, use
22  //! StopWatch. Example:
23  //!\code
24  //! StatWatch statWatch;
25  //! for (;;)
26  //! {
27  //! statWatch.start();
28  //! :
29  //! statWatch.stop();
30  //! }
31  //! size_t numSamples = statWatch.numDeltas();
32  //! unsigned long long avgTicks = statWatch.computeAvgDelta();
33  //! unsigned long long minTicks = statWatch.minDelta();
34  //! unsigned long long maxTicks = statWatch.maxDelta();
35  //!\endcode
36  //!

89 //! Compute and return the average delta. This is the average
90 //! of all recorded deltas. Also return the current minimum
91 //! and maximum deltas. The minimum delta is 0xffffffffffffffffULL
92 //! if no deltas have been recorded. The maximum delta is zero
93 //! if no deltas have been recorded.
94 inline unsigned long long StatWatch::computeAvgDelta(unsigned long long& minDelta, unsigned long long& maxDelta) const
95 {
96  maxDelta = maxDelta_;
97  minDelta = minDelta_;
98  return numDeltas_? (deltaSum_ / numDeltas_): 0;

172 //! Record current time as stop time. Compute the start-stop delta and
173 //! accumulate the result. This operation does not reset the start time.
174 //! A subsequent consecutive stop would result in a new delta using the
175 //! same start time. For example, a sequence of start-stop-start-stop
176 //! would result in two deltas, and a sequence of start-stop-stop would
177 //! also result in two deltas.
178 inline void StatWatch::stop()
179 {
180  stopTime_ = TickTime::curTime();
181  saveDelta();
182 }
