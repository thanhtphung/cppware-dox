82

85  rsp += (sprintf(buf, "%6s%8s%6s%6s%8s%8s%11s%11s%6s\n",
86  "size", "cap", "cap0", "grow", "inUse", "wmark", "allocs", "frees", "fails"), buf);
87  rsp += (sprintf(buf, "%6s%8s%6s%6s%8s%8s%11s%11s%6s\n",
88  "----", "---", "----", "----", "-----", "-----", "------", "-----", "-----"), buf);
89 
90  // Body.
91  for (unsigned int bufSize = 0; bufSize <= maxBufSize; bufSize += 4)
92  {
93  BufPool::Stat stat(pool, bufSize);
94  if ((stat.capacity() > 0) || (stat.usagePeak() > 0))
95  {

97  bufSize,

99  stat.initialCap(),
100  stat.growthFactor(),
101  stat.numInUseBufs(),
102  stat.usagePeak(),
103  stat.numAllocs(),
104  stat.numFrees(),
105  stat.numFails()), buf);
106  }
107  }
108  }
109 
110  respond(req, rsp);
