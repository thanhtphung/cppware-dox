170 //! Remove given items. If given indices are valid, return true. Return false
171 //! otherwise. If maintainOrder is true, appropriate items will be shifted up.

174 bool RefVec::rmFromIndex(size_t startAt, size_t itemCount, bool maintainOrder)
175 {
176  bool ok;
177  size_t endAt = startAt + itemCount - 1;
178  if (endAt < numItems())

181  Vec::rmFromIndex(startAt, itemCount, maintainOrder);

183  }
184  else

186  ok = false;
187  }
