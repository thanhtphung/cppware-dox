36 //!
37 //! Atomically unlock associated mutex and wait until the condition
38 //! variable is signaled. Relock the associated mutex before return.
39 //! Wait at most <i>timeoutInMsecs</i> msecs. Return true if successful.
40 //! The associated mutex must be locked before invoking this method.
41 //! Behavior is unpredictable otherwise. Due to inconsitencies among
42 //! various implementations, do not assume this wait can be interrupted.
43 //!
44 bool CondVar::wait(unsigned int timeoutInMsecs)
45 {
46  int rc;
47  if (timeoutInMsecs == ETERNITY)
48  {
49  rc = pthread_cond_wait(&cv_, &mu_);
50  }
51 
52  else
53  {
54  struct timespec now;
55  clock_gettime(CLOCK_REALTIME, &now);
56  struct timespec then;
57  then.tv_sec = now.tv_sec + (timeoutInMsecs / MSECS_PER_SEC);
58  then.tv_nsec = now.tv_nsec + ((timeoutInMsecs%MSECS_PER_SEC) * NSECS_PER_MSEC);
59  if (then.tv_nsec >= NSECS_PER_SEC)
60  {
61  ++then.tv_sec;
62  then.tv_nsec -= NSECS_PER_SEC;
63  }
64  rc = pthread_cond_timedwait(&cv_, &mu_, &then);
65  }
66 
67  return (rc == 0);
68 }
