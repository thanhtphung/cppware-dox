15  //!
16  //! A class representing an atomic unsigned int. Implemented using intrinsics
17  //! where applicable. This is usually the most efficient way to represent numbers
18  //! which must be synchronized across threads. Reference counting is an example
19  //! of typical usage. As a convenience, AtomicWord is an alias of Atomic32 in x86
20  //! builds. Example:
21  //!\code
22  //! if (refCount_++ == 0U) //refCount_ is an Atomic32 instance
23  //! {
24  //! // code to be run by the one thread which just incremented the count from zero
25  //! }
26  //! :
27  //! if (--refCount_ == 0U) //refCount_ is an Atomic32 instance
28  //! {
29  //! // code to be run by the one thread which just decremented the count to zero

32  //!
