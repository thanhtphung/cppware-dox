64  Count countyLock(*this);
65 
66  // This command needs to be run by the WinApp messaging thread.
67  if (Thread::myId() != winApp_->messagingThreadId())

70  // Serialize individual commands at onRun().
71  size_t i = cmdIndex(req);
72  SpinSection::Lock lock(ss_[i]);
73  WPARAM wparam = reinterpret_cast<WPARAM>(this);
74  LPARAM lparam = reinterpret_cast<LPARAM>(&req);
75  int ok = PostMessageW(messagingWin_, messagingId_, wparam, lparam);
76  if (ok)
77  {
