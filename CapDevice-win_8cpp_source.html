204 //
205 // Retrieve MAC addresses matching capture devices given by pcap.
206 // Returned array is to be destroyed when done.
207 //
208 static MacAddr* getIfMacs(const pcap_if_t* pcapIf)
209 {
210  unsigned int count = 0;
211  for (const pcap_if_t* p = pcapIf; p != 0; count++, p = p->next);
212 
213  MacAddr* macs = new MacAddr[count];
214  const pcap_if_t* p = pcapIf;
215  for (unsigned int i = 0; i < count; ++i, p = p->next)
216  {
217  HANDLE* a = PacketOpenAdapter(p->name);
218  if (a != 0)
219  {
220  if (*a != INVALID_HANDLE_VALUE)
221  {
222  char buf[offsetof(pktReq_t, data) + MacAddr::RawLength];
223  pktReq_t* req = reinterpret_cast<pktReq_t*>(buf);
224  req->oid = OID_802_3_CURRENT_ADDRESS;
225  req->length = MacAddr::RawLength;
226  memset(req->data, 0, MacAddr::RawLength);
227  unsigned char doSet = 0;
228  unsigned char ok = PacketRequest(a, doSet, req);
229  if (ok)
230  {
231  macs[i] = req->data;
232  }
233  }
234  PacketCloseAdapter(a);
235  }
236  }
237 
238  return macs;
239 }

483  // Public winpcap or cppgallery winpcap?

486  pcap_findalldevs(&pcapIf, pcapErr);
487  MacAddr* ifMac = getIfMacs(pcapIf);
488  if (pcapIf != 0)
489  {
490  if (Str::compareKIN("\\Device\\NETCAP_", pcapIf->name, sizeof("\\Device\\NETCAP_") - 1) != 0)

492  s_devicePrefix = "\\Device\\NPF_";
493  }
494  }

514 }
